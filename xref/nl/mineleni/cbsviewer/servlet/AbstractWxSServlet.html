34      private static final long serialVersionUID = -5563479037661945586L;

36      /**
37       * Parse locatie uit een request. Indien waarden niet geldig zijn of
38       * ontbreken worden de defaults {@link DEFAULT_XCOORD} ,
39       * {@link DEFAULT_YCOORD} en {@link OPENLS_ZOOMSCALE_STANDAARD} gebruikt.
40       * 
41       * @param request
42       *            Het servlet request
43       * @return een {@code double[xcoord,ycoord,straal]}
44       * @throws ServletException
45       *             Als parsen is mislukt
46       */
47      protected double[] parseLocation(HttpServletRequest request)
48              throws ServletException {
49          try {
50              // request params uitlezen voor het zoeken
51              final double xcoord = Double.valueOf((null == request
52                      .getParameter(REQ_PARAM_XCOORD.code) ? DEFAULT_XCOORD
53                      .toString() : request.getParameter(REQ_PARAM_XCOORD.code)));
54              final double ycoord = Double.valueOf((null == request
55                      .getParameter(REQ_PARAM_YCOORD.code) ? DEFAULT_YCOORD
56                      .toString() : request.getParameter(REQ_PARAM_YCOORD.code)));
57              final double straal = Double
58                      .valueOf((null == request
59                              .getParameter(REQ_PARAM_STRAAL.code) ? OPENLS_ZOOMSCALE_STANDAARD
60                              .toString() : request
61                              .getParameter(REQ_PARAM_STRAAL.code)));
62  
63              LOGGER.debug("request params:" + xcoord + ":" + ycoord + " straal:"
64                      + straal);
65              return new double[] { xcoord, ycoord, straal };
66          } catch (final NumberFormatException e) {
67              LOGGER.error(
68                      "Een van de vereiste parameters kon niet geparsed worden als Double.",
69                      e);
70              throw new ServletException(e);
71          }
72      }
73  
74      /**
75       * Parse de forward parameter van een request.
76       * 
77       * @see REQ_PARAM_FORWARD
78       * @return {@code true), als parameter aanwezig en waarde "true" heeft,
79       *         anders {@code false}
80       * @param request
81       *            Het servlet request
82       */
83      protected boolean parseForward(HttpServletRequest request) {
84          final boolean b = (null == request.getParameter(REQ_PARAM_FORWARD.code) ? false
85                  : Boolean.valueOf(request.getParameter(REQ_PARAM_FORWARD.code)));
86          LOGGER.debug("b heeft waarde:" + b + " voor request waarde "
87                  + request.getParameter(REQ_PARAM_FORWARD.code));
88          return b;
89      }
