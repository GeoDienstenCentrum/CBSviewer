66  	/**
67  	 * zorgt ervoor dat eventuele doubles als integer worden gerenderd. Als het
68  	 * niet lukt wordt de oorspronkelijke waarde teruggeven.
69  	 * 
70  	 * @param coord
71  	 *            een coordinaat waarde
72  	 * @return de waarde in integer formaat
73  	 */
74  	private String formatCoord(final String coord) {
75  		/** coordinaten formatter. */
76  		final DecimalFormat fmt = new DecimalFormat("###");
77  		try {
78  			// formatting als int
79  			return "" + (fmt.parse(coord).intValue());
80  		} catch (final ParseException e) {
81  			LOGGER.warn("Fout tijden parsen van coordinaat: " + coord, e);
82  			return coord;
83  		} catch (final NullPointerException e) {
84  			LOGGER.warn("Fout tijden parsen van coordinaat: " + coord, e);
85  			return coord;
86  		}
87  	}

111 	/**
112 	 * Return json van adreszoek resultaten.
113 	 * 
114 	 * @param addrl
115 	 *            Een {@code List} met adressen, de lijst mag leeg zijn
116 	 * @return een javascript array met geserialiseerde
117 	 *         {@link nl.mineleni.cbsviewer.servlet.gazetteer.lusclient.OpenLSClientAddress}
118 	 *         objecten
119 	 */
120 	private String returnJson(final List<OpenLSClientAddress> addrl) {
121 		if (addrl.size() < 1) {
122 			return "[]";
123 		} else {
124 			String json = new JSONSerializer().exclude("class")
125 					.transform(new NumberTransformer(), "xCoord", "yCoord")
126 					.prettyPrint(LOGGER.isDebugEnabled()).serialize(addrl);
127 			LOGGER.debug("json: " + json);
128 			return json;
129 		}
130 	}

139 	@Override
140 	protected void service(final HttpServletRequest request,
141 			final HttpServletResponse response) throws ServletException,
142 			IOException {
143 
144 		final boolean forwardResponse = this.parseForward(request);
145 		final String format = request.getParameter("format");
146 
147 		final String zoekTerm = request.getParameter("adres");
148 		LOGGER.debug("Zoeken naar: " + zoekTerm);
149 		if (zoekTerm.length() < 1) {
150 			request.setAttribute(REQ_PARAM_GEVONDEN.code,
151 					_RESOURCES.getString("KEY_ZOEKEN_NIETS_INGEVULD"));
152 			if (forwardResponse) {
153 				request.getRequestDispatcher("/index.jsp").forward(request,
154 						response);
155 			}
156 		} else {
157 			final Map<String, String> openLSParams = new TreeMap<>();
158 			openLSParams.put(OPENLS_REQ_PARAM_REQUEST.code,
159 					OPENLS_REQ_VALUE_GEOCODE.code);
160 			openLSParams.put(OPENLS_REQ_PARAM_SEARCH.code, zoekTerm);
161 			final GeocodeResponse gcr = this.openLSClient.doGetOpenLSRequest(
162 					this.openLSServerUrl, openLSParams);
163 			final List<OpenLSClientAddress> addrl = OpenLSClientUtil
164 					.getOpenLSClientAddressList(gcr, openLSmaxResults);
165 
166 			if (forwardResponse) {
167 				if (addrl.isEmpty()) {
168 					// niets gevonden
169 					LOGGER.debug(_RESOURCES.getString("KEY_ZOEKEN_GEEN_ADRES")
170 							+ zoekTerm);
171 					request.setAttribute(REQ_PARAM_GEVONDEN.code,
172 							_RESOURCES.getString("KEY_ZOEKEN_GEEN_ADRES"));
173 				} else if (addrl.size() == 1) {
174 					// 1 adres gevonden
175 					LOGGER.debug("Er is 1 match gevonden voor: " + zoekTerm);
176 					final OpenLSClientAddress addr = addrl.get(0);
177 
178 					request.setAttribute(REQ_PARAM_XCOORD.code,
179 							this.formatCoord(addr.getxCoord()));
180 					request.setAttribute(REQ_PARAM_YCOORD.code,
181 							this.formatCoord(addr.getyCoord()));
182 					request.setAttribute(REQ_PARAM_STRAAL.code,
183 							addr.getRadius());
184 					request.setAttribute(REQ_PARAM_GEVONDEN.code,
185 							_RESOURCES.getString("KEY_ZOEKEN_EEN_ADRES") + " "
186 									+ addr.getAddressString());
187 				} else {
188 					// meer dan 1 adressen gevonden
189 					LOGGER.debug("Er is meer dan 1 match gevonden voor: "
190 							+ zoekTerm);
191 					request.setAttribute("adreslijst", addrl);
192 					request.setAttribute(REQ_PARAM_GEVONDEN.code,
193 							_RESOURCES.getString("KEY_ZOEKEN_MEER_ADRES"));
194 				}
195 				request.getRequestDispatcher("/index.jsp").forward(request,
196 						response);
197 			} else {
198 				// in geval van directe aanroep voor bijv. ajax
199 				if (format != null && format.equalsIgnoreCase("json")) {
200 					// geeft zoekresultaat als json
201 					final String json = returnJson(addrl);
202 					response.setCharacterEncoding("UTF-8");
203 					response.setContentType("application/json");
204 					response.setContentLength(json.length());
205 					response.getWriter().print(json);
206 					response.flushBuffer();
207 				}
208 			}
209 		}
