102     /**
103      * Initialize variables called when context is initialized. Leest de waarden
104      * van {@link #ALLOWED_HOSTS} (verplichte optie) en {@link #FORCE_XML_MIME}
105      * uit de configuratie.
106      * 
107      * @param config
108      *            the <code>ServletConfig</code> object that contains
109      *            configutation information for this servlet
110      * @throws ServletException
111      *             if an exception occurs that interrupts the servlet's normal
112      *             operation
113      */
114     @Override
115     public void init(ServletConfig config) throws ServletException {
116         super.init(config);
117         final String forceXML = config.getInitParameter(FORCE_XML_MIME);
118         this.forceXmlResponse = (null != forceXML ? Boolean
119                 .parseBoolean(forceXML) : false);
120 
121         String csvHostnames = config.getInitParameter(ALLOWED_HOSTS);
122         if (csvHostnames == null) {
123             LOGGER.error(ERR_MSG_MISSING_CONFIG);
124             throw new ServletException(ERR_MSG_MISSING_CONFIG);
125         }
126         // clean-up whitespace and case the names
127         csvHostnames = csvHostnames.replaceAll("\\s", "").toLowerCase();
128         final String[] names = csvHostnames.split(";");
129         for (final String name : names) {
130             this.allowedHosts.add(name);
131             LOGGER.debug("toevoegen aan allowed host namen: " + name);
132         }
133     }

135     /**
136      * Process the HTTP Get request. Voor een Openlayers proxy wordt de url dan
137      * iets van: <br/>
138      * <code>WMSproxy/proxy?</code> <br/>
139      * waarin WMSproxy de naam van de webapp is en proxy de servlet mapping.
140      * 
141      * @param request
142      *            the request
143      * @param response
144      *            the response
145      * @throws ServletException
146      *             the servlet exception
147      */
148     @Override
149     protected void doGet(HttpServletRequest request,
150             HttpServletResponse response) throws ServletException {
151         try {
152             String serverUrl = request.getQueryString();
153             serverUrl = URLDecoder.decode(serverUrl, "UTF-8");
154             if (serverUrl.startsWith("http://")
155                     || serverUrl.startsWith("https://")) {
156                 // check if allowed
157                 if (!this.checkUrlAllowed(serverUrl)) {
158                     LOGGER.warn(serverUrl + ERR_MSG_FORBIDDEN);
159                     response.sendError(SC_FORBIDDEN, serverUrl
160                             + ERR_MSG_FORBIDDEN);
161                     response.flushBuffer();
162                 } else {
163                     LOGGER.debug("GET param:" + serverUrl);
164                     final HttpClient client = new DefaultHttpClient();
165                     if (null != this.proxyHost) {
166                         client.getParams().setParameter(
167                                 ConnRoutePNames.DEFAULT_PROXY,
168                                 new HttpHost(this.proxyHost, this.proxyPort,
169                                         "http"));
170                     }
171                     // override referer
172                     client.getParams().setParameter("Referer", "/");
173 
174                     final HttpResponse getResp = client.execute(new HttpGet(
175                             serverUrl));
176 
177                     // dump response to out
178                     if (getResp.getStatusLine().getStatusCode() == SC_OK) {
179                         // force the response to have XML content type (WMS
180                         // servers generally don't)
181                         if (this.forceXmlResponse) {
182                             response.setContentType("text/xml");
183                         }
184                         final String responseBody = EntityUtils.toString(
185                                 getResp.getEntity()).trim();
186                         response.setContentLength(responseBody.length());
187 
188                         LOGGER.debug("responseBody:" + responseBody);
189                         final PrintWriter out = response.getWriter();
190                         out.print(responseBody);
191                         response.flushBuffer();
192                     } else {
193                         LOGGER.warn("Onverwachte fout(server url=" + serverUrl
194                                 + "): " + getResp.getStatusLine().toString());
195                         response.sendError(getResp.getStatusLine()
196                                 .getStatusCode(), getResp.getStatusLine()
197                                 .toString());
198                     }
199                     client.getConnectionManager().shutdown();
200                 }
201             } else {
202                 throw new ServletException("Only HTTP(S) protocol supported");
203             }
204         } catch (final UnsupportedEncodingException e) {
205             LOGGER.error("Proxy fout.", e);
206             throw new ServletException(e);
207         } catch (final IOException e) {
208             LOGGER.error("Proxy IO fout.", e);
209             throw new ServletException(e);
210         }
211     }

282     /**
283      * Parse out the server name and check if the specified server name is in
284      * the list of allowed servernames.
285      * 
286      * @param serverUrl
287      *            the url to check
288      * @return <code>true</code> if the name of the server is found in the list
289      */
290     private boolean checkUrlAllowed(String serverUrl) {
291         serverUrl = serverUrl.toLowerCase();
292         serverUrl = serverUrl.substring(serverUrl.indexOf("//") + 2);
293         if (serverUrl.contains("/")) {
294             serverUrl = serverUrl.substring(0, serverUrl.indexOf("/"));
295         }
296         LOGGER.debug("test server = " + serverUrl);
297         return (this.allowedHosts.contains(serverUrl));
298     }
