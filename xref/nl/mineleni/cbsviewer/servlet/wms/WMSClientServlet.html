171 	/**
172 	 * Teken een schaalbalk in de (kaart) afbeelding. meter based CRS only.
173 	 * 
174 	 * @param bbox
175 	 *            the bbox
176 	 * @param image
177 	 *            afbeelding waarin de schaalbalk wordt getekend
178 	 * 
179 	 * @todo only works for CRS in meters
180 	 */
181 	private void drawScaleBar(final BufferedImage image, final BoundingBox bbox) {
182 		// CHECKSTYLE.OFF: MagicNumber - pixel layout
183 		// start positie in px
184 		final int xOffset = 10;
185 		final int fontSize = 12;
186 		final int yOffset = MAP_DIMENSION - xOffset;
187 		final Graphics2D g = image.createGraphics();
188 		g.setRenderingHint(RenderingHints.KEY_ANTIALIASING,
189 				RenderingHints.VALUE_ANTIALIAS_ON);
190 		g.setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING,
191 				RenderingHints.VALUE_TEXT_ANTIALIAS_ON);
192 
193 		// scale crs units per px
194 		final double scale = bbox.getWidth() / MAP_DIMENSION;
195 		// max lengte in px van schaalbalk
196 		int barLength = MAP_DIMENSION / 2;
197 		int dist = (int) (barLength * scale); // crs units
198 		// logaritmisch lengte afronden
199 		final int digits = (int) (Math.log(dist) / Math.log(10));
200 		final double pow10 = Math.pow(10, digits);
201 		final int rounded = (int) (dist / pow10);
202 		int barLen = 1;
203 		if (rounded > 5) {
204 			barLen = 5;
205 		} else if (rounded > 2) {
206 			barLen = 2;
207 		}
208 		dist = (int) (barLen * pow10);
209 		barLength = (int) (dist / scale);
210 
211 		// TODO CRS units gebruiken
212 		// bbox.getCoordinateReferenceSystem().getCoordinateSystem().getAxis(0).getUnit()...
213 		String units = "m";
214 		if (dist >= 1000) {
215 			dist = dist / 1000;
216 			units = "km";
217 		}
218 
219 		g.setColor(Color.BLACK);
220 		g.setStroke(new BasicStroke(2, BasicStroke.CAP_ROUND,
221 				BasicStroke.JOIN_ROUND));
222 		g.drawLine(xOffset, yOffset, xOffset + barLength, yOffset);
223 		g.drawLine(xOffset, yOffset, xOffset, yOffset - fontSize);
224 		g.drawLine(xOffset + barLength, yOffset, xOffset + barLength,
225 				yOffset - fontSize);
226 
227 		final Font font = new Font(Font.SANS_SERIF, Font.BOLD, fontSize);
228 		final FontMetrics metrics = g.getFontMetrics(font);
229 		g.setFont(font);
230 		g.drawString(
231 				dist + units,
232 				(xOffset + barLength / 2) - (metrics.stringWidth(dist + units) / 2),
233 				yOffset - metrics.getDescent() - 2);
234 		// CHECKSTYLE.ON: MagicNumber
235 	}

238 	 * Achtergrondkaart ophalen en opslaan in de cache.
239 	 * 
240 	 * @param bbox
241 	 *            the bbox
242 	 * @param baseMapType
243 	 *            het type basemap, {@code "luchtfoto"} of (default)
244 	 *            {@code "topografie"}
245 	 * @return background/basemap image
246 	 * @throws ServletException
247 	 *             Geeft aan dat er een fout is opgetreden bij het benaderen van
248 	 *             de achtergrondgrond WMS service
249 	 */
250 	private BufferedImage getBackGroundMap(final BoundingBox bbox,
251 			final String baseMapType) throws ServletException {
252 
253 		GetMapRequest map;
254 		switch (baseMapType.toLowerCase()) {
255 		case "luchtfoto":
256 			if (this.bgWMSLuFoCache.containsKey(bbox)) {
257 				// ophalen uit cache
258 				LOGGER.debug("Achtergrond " + baseMapType
259 						+ " afbeelding uit de cache serveren.");
260 				return this.bgWMSLuFoCache.getImage(bbox);
261 			}
262 			map = this.lufoWMS.createGetMapRequest();
263 			if (this.lufoWMSlayers != null) {
264 				for (final String lyr : this.lufoWMSlayers) {
265 					// per laag toevoegen met de default style
266 					map.addLayer(lyr, "");
267 				}
268 			} else {
269 				// alle lagen toevoegen
270 				for (final Layer layer : WMSUtils.getNamedLayers(this.lufoWMS
271 						.getCapabilities())) {
272 					map.addLayer(layer);
273 				}
274 			}
275 			break;
276 		case "topografie":
277 			// implicit fall thru naar default
278 		default:
279 			if (this.bgWMSCache.containsKey(bbox)) {
280 				// ophalen uit cache
281 				LOGGER.debug("Achtergrond " + baseMapType
282 						+ " afbeelding uit de cache serveren.");
283 				return this.bgWMSCache.getImage(bbox);
284 			}
285 			map = this.bgWMS.createGetMapRequest();
286 			if (this.bgWMSlayers != null) {
287 				for (final String lyr : this.bgWMSlayers) {
288 					// per laag toevoegen met de default style
289 					map.addLayer(lyr, "");
290 				}
291 			} else {
292 				// alle lagen toevoegen
293 				for (final Layer layer : WMSUtils.getNamedLayers(this.bgWMS
294 						.getCapabilities())) {
295 					map.addLayer(layer);
296 				}
297 			}
298 		}
299 
300 		map.setFormat("image/png");
301 		map.setDimensions(MAP_DIMENSION, MAP_DIMENSION);
302 		map.setTransparent(true);
303 		map.setBGColour("0xffffff");
304 		map.setExceptions("application/vnd.ogc.se_inimage");
305 		map.setSRS("EPSG:28992");
306 		map.setBBox(bbox);
307 
308 		LOGGER.debug("Achtergrond WMS url is: " + map.getFinalURL());
309 
310 		try {
311 			final GetMapResponse response = this.bgWMS.issueRequest(map);
312 			final BufferedImage image = ImageIO.read(response.getInputStream());
313 			switch (baseMapType.toLowerCase()) {
314 			case "luchtfoto":
315 				this.bgWMSLuFoCache.put(bbox, image, SECONDS_TO_CACHE_ELEMENTS);
316 				break;
317 			case "topografie":
318 				// implicit fall thru naar default
319 			default:
320 				this.bgWMSCache.put(bbox, image, SECONDS_TO_CACHE_ELEMENTS);
321 				break;
322 			}
323 
324 			if (LOGGER.isDebugEnabled()) {
325 				// achtergrond plaatje bewaren in debug modus
326 				final File temp = File.createTempFile(
327 						"bgwms",
328 						".png",
329 						new File(this.getServletContext().getRealPath(
330 								MAP_CACHE_DIR.code)));
331 				temp.deleteOnExit();
332 				ImageIO.write(image, "png", temp);
333 			}
334 
335 			return image;
336 		} catch (ServiceException | IOException e) {
337 			LOGGER.error(
338 					"Er is een fout opgetreden bij het benaderen van de achtergrond WMS service.",
339 					e);
340 			throw new ServletException(e);
341 		}
342 
343 	}

366 			return fgWMS;

370 	/**
371 	 * Haalt de feature info op.
372 	 * 
373 	 * @param bbox
374 	 *            the bbox
375 	 * @param lyrDesc
376 	 *            de layerdescriptor met de WMS informatie
377 	 * @return Een string met feature info
378 	 * @throws ServiceException
379 	 *             Geeft aan dat er een fout is opgetreden tijden het benaderen
380 	 *             van de WMS
381 	 * @throws IOException
382 	 *             Signals that an I/O exception has occurred.
383 	 */
384 	private String getFeatureInfo(final BoundingBox bbox,
385 			final LayerDescriptor lyrDesc) throws ServiceException, IOException {
386 		final BboxLayerCacheKey key = new BboxLayerCacheKey(bbox, lyrDesc);
387 		if (this.featInfoCache.containsKey(key)) {
388 			// ophalen uit cache
389 			final CachableString fInfo = this.featInfoCache.get(key);
390 			if (null != fInfo) {
391 				// dit kan null zijn in het geval het item verlopen is
392 				LOGGER.debug("FeatureInfo uit de cache serveren.");
393 				return fInfo.getItem();
394 			}
395 		}
396 
397 		try {
398 			final GetFeatureInfoRequest getFeatureInfoRequest = this
399 					.getCachedWMS(lyrDesc).createGetFeatureInfoRequest(
400 							this.getMapRequest);
401 
402 			final String[] layerNames = lyrDesc.getLayers().split(",\\s*");
403 			final Set<Layer> queryLayers = new HashSet<>();
404 			final WMSCapabilities caps = this.getCachedWMS(lyrDesc)
405 					.getCapabilities();
406 
407 			for (final Layer wmsLyr : caps.getLayerList()) {
408 				if ((wmsLyr.getName() != null)
409 						&& (wmsLyr.getName().length() != 0)) {
410 					for (final String layerName : layerNames) {
411 						if (wmsLyr.getName().equalsIgnoreCase(layerName)) {
412 							queryLayers.add(wmsLyr);
413 						}
414 					}
415 				}
416 			}
417 			getFeatureInfoRequest.setQueryLayers(queryLayers);
418 			getFeatureInfoRequest.setInfoFormat("application/vnd.ogc.gml");
419 			getFeatureInfoRequest.setFeatureCount(10);
420 			getFeatureInfoRequest.setQueryPoint(MAP_DIMENSION_MIDDLE,
421 					MAP_DIMENSION_MIDDLE);
422 			LOGGER.debug("WMS feature info request url is: "
423 					+ getFeatureInfoRequest.getFinalURL());
424 			final GetFeatureInfoResponse response = this.getCachedWMS(lyrDesc)
425 					.issueRequest(getFeatureInfoRequest);
426 
427 			final String html = FeatureInfoResponseConverter
428 					.convertToHTMLTable(response.getInputStream(), "GMLTYPE",
429 							lyrDesc);
430 			this.featInfoCache.put(key,
431 					new CachableString(html, System.currentTimeMillis()
432 							+ MILLISECONDS_TO_CACHE_ELEMENTS));
433 			return html;
434 
435 		} catch (final UnsupportedOperationException u) {
436 			LOGGER.warn("De WMS server ("
437 					+ this.getCachedWMS(lyrDesc).getInfo().getTitle()
438 					+ ") ondersteund geen GetFeatureInfoRequest.", u);
439 			return "";
440 		}
441 	}

443 	/**
444 	 * voorgrondkaart ophalen.
445 	 * 
446 	 * @param bbox
447 	 *            the bbox
448 	 * @param lyrDesc
449 	 *            de layerdescriptor met de WMS informatie
450 	 * @return voorgrond afbeelding
451 	 * @throws ServletException
452 	 *             Geeft aan dat er een fout is opgetreden bij het benaderen van
453 	 *             de voorgrond WMS service
454 	 */
455 	private BufferedImage getForeGroundMap(final BoundingBox bbox,
456 			final LayerDescriptor lyrDesc) throws ServletException {
457 
458 		final BboxLayerCacheKey key = new BboxLayerCacheKey(bbox, lyrDesc);
459 		if (this.fgWMSCache.containsKey(key)) {
460 			// ophalen uit cache
461 			final CacheImage cImg = this.fgWMSCache.get(key);
462 			if (null != cImg) {
463 				LOGGER.debug("Voorgrond afbeelding uit de cache serveren.");
464 				return cImg.getImage();
465 			}
466 		}
467 
468 		// wms request doen
469 		try {
470 			this.getMapRequest = this.getCachedWMS(lyrDesc)
471 					.createGetMapRequest();
472 			final String[] layerNames = lyrDesc.getLayers().split(",\\s*");
473 			final String[] styleNames = lyrDesc.getStyles().split(",\\s*");
474 
475 			for (int l = 0; l < layerNames.length; l++) {
476 				this.getMapRequest.addLayer(layerNames[l], styleNames[l]);
477 			}
478 			this.getMapRequest.setFormat("image/png");
479 			this.getMapRequest.setDimensions(MAP_DIMENSION, MAP_DIMENSION);
480 			this.getMapRequest.setTransparent(true);
481 			this.getMapRequest.setSRS("EPSG:28992");
482 			this.getMapRequest.setBBox(bbox);
483 			this.getMapRequest.setExceptions("application/vnd.ogc.se_inimage");
484 			this.getMapRequest.setBGColour("0xffffff");
485 			LOGGER.debug("Voorgrond WMS url is: "
486 					+ this.getMapRequest.getFinalURL());
487 
488 			// thema/voorgrond ophalen
489 			final GetMapResponse response = this.getCachedWMS(lyrDesc)
490 					.issueRequest(this.getMapRequest);
491 			final BufferedImage image = ImageIO.read(response.getInputStream());
492 			this.drawScaleBar(image, bbox);
493 			this.fgWMSCache.put(key, new CacheImage(image,
494 					SECONDS_TO_CACHE_ELEMENTS));
495 
496 			if (LOGGER.isDebugEnabled()) {
497 				// voorgrond plaatje bewaren in debug modus
498 				final File temp = File.createTempFile(
499 						"fgwms",
500 						".png",
501 						new File(this.getServletContext().getRealPath(
502 								MAP_CACHE_DIR.code)));
503 				temp.deleteOnExit();
504 				ImageIO.write(image, "png", temp);
505 			}
506 			return image;
507 		} catch (ServiceException | IOException e) {
508 			LOGGER.error(
509 					"Er is een fout opgetreden bij het benaderen van de voorgrond WMS service.",
510 					e);
511 			throw new ServletException(e);
512 		}
513 	}

538 	/**
539 	 * get legenda afbeeldingen door gebruik te maken legendUrl.
540 	 * 
541 	 * @param lyrDesc
542 	 *            de layerdescriptor met de WMS informatie
543 	 * @return een array met legenda afbeeldings bestanden
544 	 * @throws ServiceException
545 	 *             Geeft aan dat er een fout is opgetreden tijden het benaderen
546 	 *             van de WMS
547 	 * @throws IOException
548 	 *             Signals that an I/O exception has occurred.
549 	 */
550 	private File[] getLegendsFromLayerStyles(final LayerDescriptor lyrDesc)
551 			throws ServiceException, IOException {
552 		final String[] layerNames = lyrDesc.getLayers().split(",\\s*");
553 		final String[] styleNames = lyrDesc.getStyles().split(",\\s*");
554 		final File[] legends = new File[layerNames.length];
555 
556 		for (int l = 0; l < layerNames.length; l++) {
557 			final String key = layerNames[l] + "::" + styleNames[l];
