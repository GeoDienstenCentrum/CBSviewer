157 	/**
158 	 * Achtergrondkaart ophalen en opslaan in de cache.
159 	 * 
160 	 * @param bbox
161 	 *            the bbox
162 	 * @param type
163 	 *            the type
164 	 * @return background/basemap image
165 	 * @throws ServletException
166 	 *             Geeft aan dat er een fout is opgetreden bij het benaderen van
167 	 *             de achtergrondgrond WMS service
168 	 */
169 	private BufferedImage getBackGroundMap(final BoundingBox bbox,
170 			final BasemapType type) throws ServletException {
171 
172 		GetMapRequest map;
173 		switch (type) {
174 		case luchtfoto:
175 			if (this.bgWMSLuFoCache.containsKey(bbox)) {
176 				// ophalen uit cache
177 				LOGGER.debug("Achtergrond " + type
178 						+ " afbeelding uit de cache serveren.");
179 				return this.bgWMSLuFoCache.getImage(bbox);
180 			}
181 			map = this.lufoWMS.createGetMapRequest();
182 			if (this.lufoWMSlayers != null) {
183 				for (final String lyr : this.lufoWMSlayers) {
184 					// per laag toevoegen met de default style
185 					map.addLayer(lyr, "");
186 				}
187 			} else {
188 				// alle lagen toevoegen
189 				for (final Layer layer : WMSUtils.getNamedLayers(this.lufoWMS
190 						.getCapabilities())) {
191 					map.addLayer(layer);
192 				}
193 			}
194 			break;
195 		case topografie:
196 			// implicit fall thru naar default
197 		default:
198 			if (this.bgWMSCache.containsKey(bbox)) {
199 				// ophalen uit cache
200 				LOGGER.debug("Achtergrond " + type
201 						+ " afbeelding uit de cache serveren.");
202 				return this.bgWMSCache.getImage(bbox);
203 			}
204 			map = this.bgWMS.createGetMapRequest();
205 			if (this.bgWMSlayers != null) {
206 				for (final String lyr : this.bgWMSlayers) {
207 					// per laag toevoegen met de default style
208 					map.addLayer(lyr, "");
209 				}
210 			} else {
211 				// alle lagen toevoegen
212 				for (final Layer layer : WMSUtils.getNamedLayers(this.bgWMS
213 						.getCapabilities())) {
214 					map.addLayer(layer);
215 				}
216 			}
217 		}
218 
219 		map.setFormat("image/png");
220 		map.setDimensions(MAP_DIMENSION, MAP_DIMENSION);
221 		map.setTransparent(true);
222 		map.setBGColour("0xffffff");
223 		map.setExceptions("application/vnd.ogc.se_inimage");
224 		map.setSRS("EPSG:28992");
225 		map.setBBox(bbox);
226 
227 		LOGGER.debug("Achtergrond WMS url is: " + map.getFinalURL());
228 
229 		try {
230 			final GetMapResponse response = this.bgWMS.issueRequest(map);
231 			final BufferedImage image = ImageIO.read(response.getInputStream());
232 			switch (type) {
233 			case luchtfoto:
234 				this.bgWMSLuFoCache.put(bbox, image, SECONDS_TO_CACHE_ELEMENTS);
235 
236 				break;
237 			case topografie:
238 			default:
239 				this.bgWMSCache.put(bbox, image, SECONDS_TO_CACHE_ELEMENTS);
240 				break;
241 			}
242 
243 			if (LOGGER.isDebugEnabled()) {
244 				// achtergrond plaatje bewaren in debug modus
245 				final File temp = File.createTempFile(
246 						"bgwms",
247 						".png",
248 						new File(this.getServletContext().getRealPath(
249 								MAP_CACHE_DIR.code)));
250 				temp.deleteOnExit();
251 				ImageIO.write(image, "png", temp);
252 			}
253 
254 			return image;
255 		} catch (ServiceException | IOException e) {
256 			LOGGER.error(
257 					"Er is een fout opgetreden bij het benaderen van de achtergrond WMS service.",
258 					e);
259 			throw new ServletException(e);
260 		}
261 
262 	}

265 	 * zoekt of maakt de gevraagde WebMapServer.
266 	 * 
267 	 * @param lyrDesc
268 	 *            de layerdescriptor met de WMS informatie
269 	 * @return the cached wms
270 	 * @throws ServiceException
271 	 *             the service exception
272 	 * @throws IOException
273 	 *             Signals that an I/O exception has occurred.
274 	 */
275 	private WebMapServer getCachedWMS(final LayerDescriptor lyrDesc)
276 			throws ServiceException, IOException {
277 		if (this.wmsServersCache.containsKey(lyrDesc.getUrl())) {
278 			LOGGER.debug("WMS gevonden in cache.");
279 			return this.wmsServersCache.get(lyrDesc.getUrl());
280 		} else {
281 			LOGGER.debug("Aanmaken van nieuwe WMS (inclusief versie onderhandeling).");
282 			final WebMapServer fgWMS = new WebMapServer(new URL(
283 					lyrDesc.getUrl()));
284 			this.wmsServersCache.put(lyrDesc.getUrl(), fgWMS);
285 			return fgWMS;
286 		}
287 	}

289 	/**
290 	 * Haalt de feature info op.
291 	 * 
292 	 * @param bbox
293 	 *            the bbox
294 	 * @param lyrDesc
295 	 *            de layerdescriptor met de WMS informatie
296 	 * @return Een string met feature info
297 	 * @throws ServiceException
298 	 *             Geeft aan dat er een fout is opgetreden tijden het benaderen
299 	 *             van de WMS
300 	 * @throws IOException
301 	 *             Signals that an I/O exception has occurred.
302 	 */
303 	private String getFeatureInfo(final BoundingBox bbox,
304 			final LayerDescriptor lyrDesc) throws ServiceException, IOException {
305 		final BboxLayerCacheKey key = new BboxLayerCacheKey(bbox, lyrDesc);
306 		if (featInfoCache.containsKey(key)) {
307 			LOGGER.debug("FeatureInfo uit de cache serveren.");
308 			// ophalen uit cache
309 			return featInfoCache.get(key).getItem();
310 		}
311 
312 		try {
313 			final GetFeatureInfoRequest getFeatureInfoRequest = this
314 					.getCachedWMS(lyrDesc).createGetFeatureInfoRequest(
315 							this.getMapRequest);
316 
317 			final String[] layerNames = lyrDesc.getLayers().split(",\\s*");
318 			final Set<Layer> queryLayers = new HashSet<Layer>();
319 			final WMSCapabilities caps = this.getCachedWMS(lyrDesc)
320 					.getCapabilities();
321 
322 			for (final Layer wmsLyr : caps.getLayerList()) {
323 				if ((wmsLyr.getName() != null)
324 						&& (wmsLyr.getName().length() != 0)) {
325 					for (final String layerName : layerNames) {
326 						if (wmsLyr.getName().equalsIgnoreCase(layerName)) {
327 							queryLayers.add(wmsLyr);
328 						}
329 					}
330 				}
331 			}
332 			getFeatureInfoRequest.setQueryLayers(queryLayers);
333 			getFeatureInfoRequest.setInfoFormat("application/vnd.ogc.gml");
334 			getFeatureInfoRequest.setFeatureCount(10);
335 			getFeatureInfoRequest.setQueryPoint(MAP_DIMENSION_MIDDLE,
336 					MAP_DIMENSION_MIDDLE);
337 			LOGGER.debug("WMS feature info request url is: "
338 					+ getFeatureInfoRequest.getFinalURL());
339 			final GetFeatureInfoResponse response = this.getCachedWMS(lyrDesc)
340 					.issueRequest(getFeatureInfoRequest);
341 
342 			final String html = FeatureInfoResponseConverter
343 					.convertToHTMLTable(response.getInputStream(),
344 							FeatureInfoResponseConverter.Type.GMLTYPE, lyrDesc
345 									.getAttributes().split(",\\s*"));
346 			featInfoCache.put(key,
347 					new CachableString(html, System.currentTimeMillis()
348 							+ MILLISECONDS_TO_CACHE_ELEMENTS));
349 			return html;
350 
351 		} catch (final UnsupportedOperationException u) {
352 			LOGGER.warn("De WMS server ("
353 					+ this.getCachedWMS(lyrDesc).getInfo().getTitle()
354 					+ ") ondersteund geen GetFeatureInfoRequest.", u);
355 			return "";
356 		}
357 	}

359 	/**
360 	 * voorgrondkaart ophalen.
361 	 * 
362 	 * @param bbox
363 	 *            the bbox
364 	 * @param lyrDesc
365 	 *            de layerdescriptor met de WMS informatie
366 	 * @return voorgrond afbeelding
367 	 * @throws ServletException
368 	 *             Geeft aan dat er een fout is opgetreden bij het benaderen van
369 	 *             de voorgrond WMS service
370 	 */
371 	private BufferedImage getForeGroundMap(final BoundingBox bbox,
372 			final LayerDescriptor lyrDesc) throws ServletException {
373 
374 		final BboxLayerCacheKey key = new BboxLayerCacheKey(bbox, lyrDesc);
375 		if (fgWMSCache.containsKey(key)) {
376 			// ophalen uit cache
377 			LOGGER.debug("Voorgrond afbeelding uit de cache serveren.");
378 			return fgWMSCache.get(key).getImage();
379 		}
380 
381 		// wms request doen
382 		try {
383 			this.getMapRequest = this.getCachedWMS(lyrDesc)
384 					.createGetMapRequest();
385 			final String[] layerNames = lyrDesc.getLayers().split(",\\s*");
386 			final String[] styleNames = lyrDesc.getStyles().split(",\\s*");
387 
388 			for (int l = 0; l < layerNames.length; l++) {
389 				this.getMapRequest.addLayer(layerNames[l], styleNames[l]);
390 			}
391 			this.getMapRequest.setFormat("image/png");
392 			this.getMapRequest.setDimensions(MAP_DIMENSION, MAP_DIMENSION);
393 			this.getMapRequest.setTransparent(true);
394 			this.getMapRequest.setSRS("EPSG:28992");
395 			this.getMapRequest.setBBox(bbox);
396 			this.getMapRequest.setExceptions("application/vnd.ogc.se_inimage");
397 			this.getMapRequest.setBGColour("0xffffff");
398 			LOGGER.debug("Voorgrond WMS url is: "
399 					+ this.getMapRequest.getFinalURL());
400 
401 			// thema/voorgrond ophalen
402 			final GetMapResponse response = this.getCachedWMS(lyrDesc)
403 					.issueRequest(this.getMapRequest);
404 			final BufferedImage image = ImageIO.read(response.getInputStream());
405 
406 			fgWMSCache.put(key,
407 					new CacheImage(image, SECONDS_TO_CACHE_ELEMENTS));
408 
409 			if (LOGGER.isDebugEnabled()) {
410 				// voorgrond plaatje bewaren in debug modus
411 				final File temp = File.createTempFile(
412 						"fgwms",
413 						".png",
414 						new File(this.getServletContext().getRealPath(
415 								MAP_CACHE_DIR.code)));
416 				temp.deleteOnExit();
417 				ImageIO.write(image, "png", temp);
418 			}
419 			return image;
420 		} catch (ServiceException | IOException e) {
421 			LOGGER.error(
422 					"Er is een fout opgetreden bij het benaderen van de achtergrond WMS service.",
423 					e);
424 			throw new ServletException(e);
425 		}
426 	}

428 	/**
429 	 * Haalt de legenda op voor de thema laag.
430 	 * 
431 	 * @param lyrDesc
432 	 *            de layerdescriptor met de WMS informatie
433 	 * @return een array met legenda afbeeldings bestanden
434 	 * @throws ServiceException
435 	 *             Geeft aan dat er een fout is opgetreden tijden het benaderen
436 	 *             van de WMS
437 	 * @throws IOException
438 	 *             Signals that an I/O exception has occurred.
439 	 */
440 	private File[] getLegends(final LayerDescriptor lyrDesc)
441 			throws ServiceException, IOException {
442 
443 		final String[] layerNames = lyrDesc.getLayers().split(",\\s*");
444 		final String[] styleNames = lyrDesc.getStyles().split(",\\s*");
445 
446 		final File[] legends = new File[layerNames.length];
447 		try {
448 			final GetLegendGraphicRequest legend = this.getCachedWMS(lyrDesc)
449 					.createGetLegendGraphicRequest();
450 			BufferedImage image;
451 			for (int l = 0; l < layerNames.length; l++) {
452 				final String key = layerNames[l] + "::" + styleNames[l];
453 				if (legendCache.containsKey(key)) {
454 					// in de cache kijken of we deze legenda afbeelding al
455 					// hebben
456 					legends[l] = new File(legendCache.get(key).getName());
457 					if (!legends[l].exists()) {
458 						// (mogelijk) is het bestand gewist..
459 						ImageIO.write(legendCache.get(key).getImage(), "png",
460 								legends[l]);
461 					}
462 					LOGGER.debug("Legenda bestand uit cache: "
463 							+ legends[l].getAbsolutePath());
464 				} else {
465 					// legenda opvragen
466 					legend.setLayer(layerNames[l]);
467 					legend.setStyle(styleNames[l]);
468 					legend.setFormat("image/png");
469 					legend.setExceptions("application/vnd.ogc.se_inimage");
470 
471 					LOGGER.debug("Voorgrond WMS legenda url is: "
472 							+ legend.getFinalURL());
473 					final GetLegendGraphicResponse response = this
474 							.getCachedWMS(lyrDesc).issueRequest(legend);
475 					image = ImageIO.read(response.getInputStream());
476 					legends[l] = File.createTempFile(
477 							"legenda",
478 							".png",
479 							new File(this.getServletContext().getRealPath(
480 									MAP_CACHE_DIR.code)));
481 					legends[l].deleteOnExit();
482 					legendCache.put(key,
483 							new CacheImage(image, legends[l].getAbsolutePath(),
484 									System.currentTimeMillis()
485 											+ MILLISECONDS_TO_CACHE_ELEMENTS
486 											* 24));
487 					LOGGER.debug("Legenda bestand: "
488 							+ legends[l].getAbsolutePath());
489 					ImageIO.write(image, "png", legends[l]);
490 				}
491 			}
492 		} catch (final UnsupportedOperationException u) {
493 			LOGGER.warn("De WMS server ("
494 					+ this.getCachedWMS(lyrDesc).getInfo().getTitle()
495 					+ ") ondersteund geen GetLegendGraphicRequest.", u);
496 			return null;
497 		}
498 		return legends;
499 	}

501 	/**
502 	 * kaart maken op basis van de opgehaalde afbeeldingen.
503 	 * 
504 	 * @param imageVoorgrond
505 	 *            de voorgrondkaart
506 	 * @param imageAchtergrond
507 	 *            de achtergrondgrondkaart
508 	 * @return de file met de afbeelding
509 	 * @throws IOException
510 	 *             Signals that an I/O exception has occurred.
511 	 */
512 	private File getMap(final BufferedImage imageVoorgrond,
513 			final BufferedImage imageAchtergrond) throws IOException {
514 
515 		final BufferedImage composite = new BufferedImage(MAP_DIMENSION,
516 				MAP_DIMENSION, BufferedImage.TYPE_INT_ARGB);
517 		final Graphics2D g = composite.createGraphics();
518 
519 		g.drawImage(imageAchtergrond, 0, 0, null);
520 		if (imageVoorgrond != null) {
521 			final float[] scales = { 1f, 1f, 1f, 0.8f };
522 			final RescaleOp rop = new RescaleOp(scales, new float[4], null);
523 			g.drawImage(imageVoorgrond, rop, 0, 0);
524 			// zoeklocatie intekenen met plaatje
525 			final BufferedImage infoImage = ImageIO.read(new File(this
526 					.getClass().getClassLoader().getResource("info.png")
527 					.getFile()));
528 			// CHECKSTYLE.OFF: MagicNumber - dit zijn midden en hoogte van het
529 			// plaatje "info.png"
530 			g.drawImage(infoImage, MAP_DIMENSION_MIDDLE - 16,
531 					MAP_DIMENSION_MIDDLE - 37, null);
532 			// CHECKSTYLE.ON: MagicNumber
533 		}
534 		// opslaan van plaatje zodat de browser het op kan halen
535 		final File kaartAfbeelding = File.createTempFile(
536 				"wmscombined",
537 				".png",
538 				new File(this.getServletContext().getRealPath(
539 						MAP_CACHE_DIR.code)));
540 		kaartAfbeelding.deleteOnExit();
541 		ImageIO.write(composite, "png", kaartAfbeelding);
542 		g.dispose();
543 		return kaartAfbeelding;
544 	}

546 	/*
547 	 * (non-Javadoc)
548 	 * 
549 	 * @see
550 	 * nl.mineleni.cbsviewer.servlet.AbstractBaseServlet#init(javax.servlet.
551 	 * ServletConfig)
552 	 */
553 	@Override
554 	public void init(final ServletConfig config) throws ServletException {
555 		super.init(config);
556 		try {
557 			this.bgWMSCache = new WMSCache(this.getServletContext()
558 					.getRealPath(MAP_CACHE_DIR.code), NUMBER_CACHE_ELEMENTS);
559 		} catch (final IOException e) {
560 			LOGGER.error(
561 					"Inititalisatie fout voor de achtergrond topografie cache.",
562 					e);
563 		}
