36  public final class FeatureInfoResponseConverter {

46  	 * Cleanup html.
47  	 * 
48  	 * @param htmlStream

55  	private static String cleanupHTML(final InputStream htmlStream)

64  	/**
65  	 * Converteer gml imputstream naar html tabel.
66  	 * 
67  	 * @param gmlStream
68  	 *            the gml stream
69  	 * @param attributes
70  	 *            the attributes
71  	 * @return een html tabel
72  	 * @throws IOException
73  	 *             Signals that an I/O exception has occurred.
74  	 */
75  	private static String convertGML(final InputStream gmlStream,
76  			final String[] attributes) throws IOException {
77  		final StringBuilder sb = new StringBuilder();
78  		try {
79  			final GML gml = new GML(Version.WFS1_0);
80  			final SimpleFeatureIterator iter = gml
81  					.decodeFeatureIterator(gmlStream);
82  			if (iter.hasNext()) {
83  				// tabel maken
84  				sb.append("<table id=\"attribuutTabel\" class=\"attribuutTabel\">");
85  				sb.append("<caption>");
86  				sb.append("Informatie over de zoeklocatie.");
87  				sb.append("</caption>");
88  				sb.append("<thead><tr>");
89  				for (final String n : attributes) {
90  					sb.append("<th scope=\"col\">" + n + "</th>");
91  				}
92  				sb.append("</tr></thead>");
93  				sb.append("<tbody>");
94  				int i = 0;
95  				while (iter.hasNext()) {
96  					sb.append("<tr class=\"" + (((i++ % 2) == 0) ? "" : "even")
97  							+ "\">");
98  					final SimpleFeature f = iter.next();
99  					for (final String n : attributes) {
100 						sb.append("<td>" + f.getAttribute(n) + "</td>");
101 					}
102 					sb.append("</tr>");
103 				}
104 				sb.append("</tbody>");
105 				sb.append("</table>");
106 				iter.close();
107 				LOGGER.debug("Gemaakte HTML tabel:\n" + sb);
108 			} else {
109 				LOGGER.debug("Geen attribuut info voor deze locatie/zoomnivo");
110 				return RESOURCES.getString("KEY_INFO_GEEN_FEATURES");
111 			}
112 		} catch (ParserConfigurationException | SAXException e) {
113 			LOGGER.error("Fout tijdens parsen van GML. ", e);
114 		} finally {
115 			gmlStream.close();
116 		}
117 		return sb.toString();
118 	}

120 	/**
121 	 * Converteert een stream naar een string.
122 	 * 
123 	 * @param is
124 	 *            de InputStream met data
125 	 * @return de data als string
126 	 * @throws IOException
127 	 *             Signals that an I/O exception has occurred.
128 	 */
129 	private static String convertStreamToString(final InputStream is)
130 			throws IOException {
131 		if (is != null) {
132 			final Writer writer = new StringWriter();
133 			final char[] buffer = new char[1024];
134 			try {
135 				final Reader reader = new BufferedReader(new InputStreamReader(
136 						is, "UTF-8"));
137 				int n;
138 				while ((n = reader.read(buffer)) != -1) {
139 					writer.write(buffer, 0, n);
140 				}
141 			} finally {
142 				is.close();
143 			}
144 			return writer.toString();
145 		} else {
146 			return "";
147 		}
148 	}

150 	/**
151 	 * Converteer de input naar een html tabel.
152 	 * 
153 	 * @param input
154 	 *            inputstream met de featureinfo response.
155 	 * @param type
156 	 *            het type conversie, ondersteund is {@code "GMLTYPE"}
157 	 * @param attributes
158 	 *            namen van de feature attributen
159 	 * @return een html tabel
160 	 * @throws IOException
161 	 *             Signals that an I/O exception has occurred.
162 	 */
163 	public static String convertToHTMLTable(final InputStream input,
164 			final String type, final String[] attributes) throws IOException {
165 		switch (type.toUpperCase()) {
166 		case "GMLTYPE":
167 			return convertGML(input, attributes);
168 		case "HTMLTYPE":
169 			return cleanupHTML(input);
170 		default:
171 			return convertStreamToString(input);
172 		}
173 	}
