7   package nl.mineleni.cbsviewer.servlet;
8   
9   import java.io.IOException;
10  
11  import javax.servlet.Filter;
12  import javax.servlet.FilterChain;
13  import javax.servlet.FilterConfig;
14  import javax.servlet.ServletException;
15  import javax.servlet.ServletRequest;
16  import javax.servlet.ServletResponse;
17  import javax.servlet.http.HttpServletResponse;
18  
19  /**
20   * Stelt response headers in voor de applicatie.
21   * 
22   * @author prinsmc
23   *
24   */
25  public class ResponseHeadersFilter implements Filter {
26  
27  	/**
28  	 * Stelt extra responseheaders in.
29  	 * 
30  	 * {@inheritDoc}
31  	 * 
32  	 * @param req
33  	 *            het request
34  	 * @param res
35  	 *            de response
36  	 * @param chain
37  	 *            de filter chain
38  	 * @throws IOException
39  	 *             Signals that an I/O exception has occurred.
40  	 * @throws ServletException
41  	 *             the servlet exception
42  	 * @see javax.servlet.Filter#doFilter(javax.servlet.ServletRequest,
43  	 *      javax.servlet.ServletResponse, javax.servlet.FilterChain)
44  	 * 
45  	 * 
46  	 */
47  	@Override
48  	public void doFilter(final ServletRequest req, final ServletResponse res,
49  			final FilterChain chain) throws IOException, ServletException {
50  
51  		HttpServletResponse response = (HttpServletResponse) res;
52  		response.setHeader("X-UA-Compatible", "IE=edge");
53  		response.setHeader("Access-Control-Allow-Origin", "*");
54  		chain.doFilter(req, res);
55  	}
56  
57  	/*
58  	 * (non-Javadoc)
59  	 * 
60  	 * @see javax.servlet.Filter#init(javax.servlet.FilterConfig)
61  	 */
62  	@Override
63  	public void init(final FilterConfig filterConfig) throws ServletException {
64  		// do nothing
65  	}
66  
67  	/*
68  	 * (non-Javadoc)
69  	 * 
70  	 * @see javax.servlet.Filter#destroy()
71  	 */
72  	@Override
73  	public void destroy() {
74  		// do nothing
75  	}
76  }
