7   package nl.mineleni.cbsviewer.util;
8   
9   import org.apache.log4j.Logger;
10  import org.geotools.geometry.Envelope2D;
11  import org.geotools.referencing.CRS;
12  import org.opengis.geometry.BoundingBox;
13  import org.opengis.referencing.FactoryException;
14  import org.opengis.referencing.NoSuchAuthorityCodeException;
15  
16  /**
17   * Spatial Utilities.
18   * 
19   * @author prinsmc
20   */
21  public final class SpatialUtil {
22  	/** logger. */
23  	private static final Logger LOGGER = Logger.getLogger(SpatialUtil.class);
24  
25  	/**
26  	 * Instantiates a new spatial util.
27  	 */
28  	private SpatialUtil() {
29  	}
30  
31  	/**
32  	 * Maakt een vierkante bbox met RD CRS met lengte en hoogte van.
33  	 * 
34  	 * @param xcoord
35  	 *            xcoord
36  	 * @param ycoord
37  	 *            ycoord
38  	 * @param afstand
39  	 *            afstand
40  	 * @return the bounding box {@code 2 x afstand} (dus straal {@code afstand})
41  	 *         om het het coordinatenpaar {@code xcoord};{@code ycoord}.
42  	 */
43  	public static BoundingBox calcRDBBOX(final double xcoord,
44  			final double ycoord, final double afstand) {
45  		try {
46  			return new Envelope2D(CRS.decode("EPSG:28992"), (xcoord - afstand),
47  					(ycoord - afstand), 2 * afstand, 2 * afstand);
48  		} catch (final NoSuchAuthorityCodeException e) {
49  			LOGGER.fatal("De gevraagde CRS autoriteit is niet gevonden.", e);
50  		} catch (final FactoryException e) {
51  			LOGGER.fatal(
52  					"Gevraagde GeoTools factory voor CRS is niet gevonden.", e);
53  		}
54  		return null;
55  	}
56  }
