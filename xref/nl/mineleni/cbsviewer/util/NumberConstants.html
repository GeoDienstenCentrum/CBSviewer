7   package nl.mineleni.cbsviewer.util;
8   
9   /**
10   * Numerieke constanten voor de applicatie.
11   * 
12   * @see StringConstants
13   * @author mprins
14   * @since 1.5
15   * @note bevat de constanten en default waarden voor de applicatie
16   */
17  public enum NumberConstants {
18  	/** straal voor een adres zoekresultaat, waarde in meter. */
19  	OPENLS_ZOOMSCALE_ADRES(150),
20  	/** straal voor een postcode zoekresultaat, waarde in meter. */
21  	OPENLS_ZOOMSCALE_POSTCODE(500),
22  	/** straal voor een plaats zoekresultaat, waarde in meter. */
23  	OPENLS_ZOOMSCALE_PLAATS(3000),
24  	/** straal voor een gemeente zoekresultaat, waarde in meter. */
25  	OPENLS_ZOOMSCALE_GEMEENTE(10000),
26  	/** straal voor een provincie zoekresultaat, waarde in meter. */
27  	OPENLS_ZOOMSCALE_PROVINCIE(50000),
28  	/** straal voor een default zoekresultaat, waarde in meter. */
29  	OPENLS_ZOOMSCALE_STANDAARD(18750)/* 150000 ~ heel NL */,
30  
31  	/** default x coordinaat. */
32  	DEFAULT_XCOORD(140000),
33  	/** default y coordinaat. */
34  	DEFAULT_YCOORD(462500),
35  
36  	/** default font size, 12. */
37  	DEFAULT_FONT_SIZE(12);
38  
39  	/** de waarde van dit object. */
40  	private final Number number;
41  
42  	/**
43  	 * constructor.
44  	 * 
45  	 * @param number
46  	 *            the number
47  	 */
48  	NumberConstants(final Number number) {
49  		this.number = number;
50  	}
51  
52  	/**
53  	 * Double value van dit object.
54  	 * 
55  	 * @return de waarde van dit object als double
56  	 * @see java.lang.Number#doubleValue()
57  	 */
58  	public double doubleValue() {
59  		return this.number.doubleValue();
60  	}
61  
62  	/**
63  	 * Int value van dit object.
64  	 * 
65  	 * @return de waarde van dit object als integer
66  	 * @see java.lang.Number#intValue()
67  	 */
68  	public int intValue() {
69  		return this.number.intValue();
70  	}
71  
72  	/**
73  	 * String value van dit object.
74  	 * 
75  	 * @return de waarde van dit object als string
76  	 * @see java.lang.Enum#toString()
77  	 * @see java.lang.Number#toString()
78  	 */
79  	@Override
80  	public String toString() {
81  		return this.number.toString();
82  	}
83  }
