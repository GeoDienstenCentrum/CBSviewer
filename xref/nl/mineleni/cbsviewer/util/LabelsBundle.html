17  public class LabelsBundle {
18  	final Logger LOGGER = LoggerFactory.getLogger(LabelsBundle.class);
19  	private ResourceBundle resBundle;
20  
21  	public LabelsBundle() {
22  		try {
23  			this.resBundle = ResourceBundle
24  					.getBundle(
25  							"LabelsBundle",
26  							ResourceBundle.Control
27  									.getControl(ResourceBundle.Control.FORMAT_PROPERTIES));
28  		} catch (final NullPointerException e) {
29  			LOGGER.error("baseName of control is null.", e);
30  		} catch (final MissingResourceException e) {
31  			LOGGER.error(
32  					"De resourcebundel (LabelsBundle.properties) is niet gevonden",
33  					e);
34  		} catch (final IllegalArgumentException e) {
35  			LOGGER.error(
36  					"ResourceBundle.Control heeft geen locale kunnen achterhalen.",
37  					e);
38  		}
39  	}
40  
41  	/**
42  	 * Geeft de string voor de sleutel, de string kan leeg zijn in het geval van
43  	 * ontbreken van de key.
44  	 * 
45  	 * @param key
46  	 *            de sleutel
47  	 * @return de string
48  	 */
49  	public String getString(final String key) {
50  		String s = "";
51  		try {
52  			s = this.resBundle.getString(key);
53  		} catch (final MissingResourceException e) {
54  			LOGGER.error("Er is geen object gevonden voor de gevraagde key ("
55  					+ key + ").", e);
56  		} catch (final NullPointerException e) {
57  			LOGGER.error("Sleutel (" + key + ") is null.", e);
58  		} catch (final ClassCastException e) {
59  			LOGGER.error("Het Object voor de gevraagde sleutel  (" + key
60  					+ ") is geen String.", e);
61  		}
62  		return s;
63  	}
