30  /**
31   * Parse reverse geocode response.
32   * 
33   * @author prinsmc
34   *
35   */
36  public class OpenLSReverseGeocodeResponseParser extends AbstractOpenLSParser {
37  	/** logger. */
38  	private static final Logger LOGGER = LoggerFactory
39  			.getLogger(OpenLSReverseGeocodeResponseParser.class);
40  
41  	/*
42  	 * (non-Javadoc)
43  	 * 
44  	 * @see org.xml.sax.helpers.DefaultHandler#endElement(java.lang.String,
45  	 * java.lang.String, java.lang.String)
46  	 */
47  	@Override
48  	public void endElement(final String uri, final String localName,
49  			final String qName) throws SAXException {
50  		final String[] nsName = qName.split(":");
51  		String eName = "";
52  		if (nsName.length > 1) {
53  			eName = nsName[1];
54  		} else {
55  			eName = nsName[0];
56  		}
57  
58  		switch (eName.toLowerCase()) {
59  		case "reversegeocoderesponse":
60  			break;
61  		case "reversegeocodedlocation":
62  			final ReverseGeocodedLocation loc = (ReverseGeocodedLocation) this.objStack
63  					.pop();
64  			if (this.objStack.peek().getClass() == new ReverseGeocodeResponse()
65  					.getClass()) {
66  				((ReverseGeocodeResponse) (this.objStack.peek()))
67  						.setReverseGeocodedLocation(loc);
68  			}
69  			break;
70  		case "point":
71  			final Point point = (Point) (this.objStack.pop());
72  			if (this.objStack.peek().getClass() == new ReverseGeocodedLocation()
73  					.getClass()) {
74  				((ReverseGeocodedLocation) (this.objStack.peek()))
75  						.setPoint(point);
76  			}
77  			break;
78  		case "pos":
79  			final Pos pos = (Pos) (this.objStack.pop());
80  			pos.setXY(this.eValBuf.toString());
81  			if (this.objStack.peek().getClass() == new Point().getClass()) {
82  				((Point) (this.objStack.peek())).addPos(pos);
83  			}
84  			break;
85  		case "address":
86  			final Address address = (Address) (this.objStack.pop());
87  			if (this.objStack.peek().getClass() == new ReverseGeocodedLocation()
88  					.getClass()) {
89  				((ReverseGeocodedLocation) (this.objStack.peek()))
90  						.setAddress(address);
91  			}
92  			break;
93  		case "streetaddress":
94  			final StreetAddress streetaddress = (StreetAddress) (this.objStack
95  					.pop());
96  			if (this.objStack.peek().getClass() == new Address().getClass()) {
97  				((Address) (this.objStack.peek()))
98  						.setStreetAddress(streetaddress);
99  			}
100 			break;
101 		case "building":
102 			final Building building = (Building) (this.objStack.pop());
103 			if (this.objStack.peek().getClass() == new StreetAddress()
104 					.getClass()) {
105 				((StreetAddress) (this.objStack.peek())).setBuilding(building);
106 			}
107 			break;
108 		case "street":
109 			final Street street = (Street) (this.objStack.pop());
110 			street.setStreet(this.eValBuf.toString());
111 			if (this.objStack.peek().getClass() == new StreetAddress()
112 					.getClass()) {
113 				((StreetAddress) (this.objStack.peek())).setStreet(street);
114 			}
115 			break;
116 		case "place":
117 			final Place place = (Place) (this.objStack.pop());
118 			place.setPlace(this.eValBuf.toString());
119 			if (this.objStack.peek().getClass() == new Address().getClass()) {
120 				((Address) (this.objStack.peek())).addPlace(place);
121 			}
122 			break;
123 		case "postalcode":
124 			final PostalCode pc = (PostalCode) (this.objStack.pop());
125 			pc.setPostalCode(this.eValBuf.toString());
126 			if (this.objStack.peek().getClass() == new Address().getClass()) {
127 				((Address) (this.objStack.peek())).setPostalCode(pc);
128 			}
129 			break;
130 		case "searchcentredistance":
131 			final SearchCentreDistance dist = (SearchCentreDistance) (this.objStack
132 					.pop());
133 			if (this.objStack.peek().getClass() == new ReverseGeocodedLocation()
134 					.getClass()) {
135 				((ReverseGeocodedLocation) (this.objStack.peek()))
136 						.setSearchCentreDistance(dist);
137 			}
138 			break;
139 		default:
140 			break;
141 		}
142 	}

144 	/**
145 	 * Gets the reverse geocode response.
146 	 * 
147 	 * @return the reverse geocode response
148 	 */
149 	public ReverseGeocodeResponse getReverseGeocodeResponse() {
150 		ReverseGeocodeResponse geocodeResponse = null;
151 		if ((this.objStack.firstElement() != null)
152 				&& (this.objStack.firstElement().getClass() == new ReverseGeocodeResponse()
153 						.getClass())) {
154 			geocodeResponse = (ReverseGeocodeResponse) this.objStack
155 					.firstElement();
156 		}
157 		// LOGGER.debug(geocodeResponse.toXML());
158 		return geocodeResponse;
159 	}
160 
161 	/**
162 	 * Parses the open ls reverse geocode response.
163 	 * 
164 	 * @param data
165 	 *            the data which is an OpenLS response xml document
166 	 * @return the geocode response object, will return null if parsing the data
167 	 *         failed
168 	 */
169 	public ReverseGeocodeResponse parseOpenLSReverseGeocodeResponse(
170 			final String data) {
171 		this.objStack.clear();
172 		try {
173 			final InputSource input = new InputSource(new StringReader(data));
174 			input.setEncoding("UTF-8");
175 			this.parser.parse(input, this);
176 		} catch (final SAXException | IOException e) {
177 			LOGGER.error("OpenLS response XML verwerken is mislukt: " + data
178 					+ ": ", e);
179 		}
180 		return this.getReverseGeocodeResponse();
181 	}

183 	/*
184 	 * (non-Javadoc)
185 	 * 
186 	 * @see org.xml.sax.helpers.DefaultHandler#startElement(java.lang.String,
187 	 * java.lang.String, java.lang.String, org.xml.sax.Attributes)
188 	 */
189 	@Override
190 	public void startElement(final String uri, final String localName,
191 			final String qName, final Attributes attributes)
192 			throws SAXException {
193 		this.eValBuf = new StringBuffer();
194 		final String[] nsName = qName.split(":");
195 		String eName = nsName[0];
196 		if (nsName.length > 1) {
197 			eName = nsName[1];
198 		}
199 
200 		switch (eName.toLowerCase()) {
201 		case "reversegeocoderesponse":
202 			this.objStack.push(new ReverseGeocodeResponse());
203 			break;
204 		case "reversegeocodedlocation":
205 			this.objStack.push(new ReverseGeocodedLocation());
206 			break;
207 		case "point":
208 			final Point point = new Point();
209 			for (int i = 0; i < attributes.getLength(); i++) {
210 				final String key = attributes.getQName(i);
211 				final String value = attributes.getValue(i);
212 				if (key.equalsIgnoreCase("srsName")) {
213 					point.setSrsName(value);
214 				}
215 			}
216 			this.objStack.push(point);
217 			break;
218 		case "pos":
219 			final Pos pos = new Pos();
220 			for (int i = 0; i < attributes.getLength(); i++) {
221 				final String key = attributes.getQName(i);
222 				final String value = attributes.getValue(i);
223 				if (key.equalsIgnoreCase("dimension")) {
224 					pos.setDimension(Integer.parseInt(value));
225 				}
226 			}
227 			this.objStack.push(pos);
228 			break;
229 		case "address":
230 			final Address addr = new Address();
231 			for (int i = 0; i < attributes.getLength(); i++) {
232 				final String key = attributes.getQName(i);
233 				final String value = attributes.getValue(i);
234 				if (key.equalsIgnoreCase("countryCode")) {
235 					addr.setCountryCode(value);
236 				}
237 			}
238 			this.objStack.push(addr);
239 			break;
240 		case "streetaddress":
241 			final StreetAddress strAddr = new StreetAddress();
242 			this.objStack.push(strAddr);
243 			break;
244 		case "street":
245 			final Street street = new Street();
246 			this.objStack.push(street);
247 			break;
248 		case "place":
249 			final Place place = new Place();
250 			for (int i = 0; i < attributes.getLength(); i++) {
251 				final String key = attributes.getQName(i);
252 				final String value = attributes.getValue(i);
253 				if (key.equalsIgnoreCase("type")) {
254 					place.setType(value);
255 				}
256 			}
257 			this.objStack.push(place);
258 			break;
259 		case "searchcentredistance":
260 			final SearchCentreDistance dist = new SearchCentreDistance();
261 			this.objStack.push(dist);
262 			for (int i = 0; i < attributes.getLength(); i++) {
263 				final String key = attributes.getQName(i);
264 				final String value = attributes.getValue(i);
265 				if (key.equalsIgnoreCase("uom")) {
266 					dist.setUom(value);
267 				}
268 				if (key.equalsIgnoreCase("value")) {
269 					dist.setValue(Double.parseDouble(value));
270 				}
271 				if (key.equalsIgnoreCase("accuracy")) {
272 					dist.setAccuracy(Double.parseDouble(value));
273 				}
274 			}
275 			break;
276 		default:
277 			LOGGER.warn("Onbekend element '" + eName + "' wordt genegeerd.");
278 			break;
279 		}
280 	}
