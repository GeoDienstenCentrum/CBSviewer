7   package nl.mineleni.cbsviewer.util;
8   
9   import static org.junit.Assert.assertTrue;
10  
11  import org.geotools.geometry.GeneralEnvelope;
12  import org.junit.Test;
13  import org.opengis.geometry.BoundingBox;
14  
15  /**
16   * Testcase voor {@link SpatialUtil}.
17   * 
18   * @author prinsmc
19   * 
20   */
21  public class SpatialUtilTest {
22  
23  	/** de constante STRAAL. {@value} */
24  	private static final double STRAAL = 1500;
25  
26  	/** de constante YCOORD. {@value} */
27  	private static final double YCOORD = 469199;
28  
29  	/** de constante XCOORD. {@value} */
30  	private static final double XCOORD = 148082;
31  
32  	/**
33  	 * Test methode voor {@link SpatialUtil#calcRDBBOX(double, double, double)}.
34  	 */
35  	@Test
36  	public final void testCalcRDBBOX() {
37  		final GeneralEnvelope expected = new GeneralEnvelope(new double[] {
38  				(XCOORD - (STRAAL)), (YCOORD - (STRAAL)) }, new double[] {
39  				(XCOORD + (STRAAL)), (YCOORD + (STRAAL)) });
40  
41  		final BoundingBox actual = SpatialUtil.calcRDBBOX(XCOORD, YCOORD,
42  				STRAAL);
43  		assertTrue(expected.contains(actual, true));
44  	}
45  }
