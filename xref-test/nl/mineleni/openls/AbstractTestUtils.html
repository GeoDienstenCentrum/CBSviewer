6   import java.io.BufferedInputStream;
7   import java.io.File;
8   import java.io.FileInputStream;
9   import java.io.IOException;
10  import java.net.URL;
11  import java.util.List;
12  
13  /**
14   * AbstractTestUtils bevat utility test methoden.
15   * 
16   * @author Mark
17   */
18  public class AbstractTestUtils {
19  
20  	/**
21  	 * List directory filenames.
22  	 * 
23  	 * @param folder
24  	 *            the folder
25  	 * @param list
26  	 *            the list
27  	 */
28  	protected void listDirectoryFilenames(final File folder,
29  			final List<File> list) {
30  		folder.setReadOnly();
31  		final File[] files = folder.listFiles();
32  		for (final File file : files) {
33  			if (file.getName().endsWith(".xml")) {
34  				list.add(file);
35  			}
36  			if (file.isDirectory()) {
37  				this.listDirectoryFilenames(file, list);
38  			}
39  		}
40  	}
41  
42  	/**
43  	 * Read file as string.
44  	 * 
45  	 * @param filePath
46  	 *            the file path
47  	 * @return inhoud van de file als string
48  	 * @throws IOException
49  	 *             Signals that an I/O exception has occurred.
50  	 */
51  	protected String readFileAsString(final String filePath)
52  			throws java.io.IOException {
53  		final URL url = this.getClass().getResource(filePath);
54  		final File file = new File(url.getFile());
55  		final byte[] buffer = new byte[(int) file.length()];
56  		final BufferedInputStream f = new BufferedInputStream(
57  				new FileInputStream(file));
58  		f.read(buffer);
59  		f.close();
60  		return new String(buffer);
61  	}
62  }
