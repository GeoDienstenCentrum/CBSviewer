7   package nl.mineleni.openls.databinding.openls;
8   
9   import static org.custommonkey.xmlunit.XMLAssert.assertXMLEqual;
10  
11  import java.io.IOException;
12  
13  import nl.mineleni.openls.AbstractTestUtils;
14  import nl.mineleni.openls.databinding.gml.Point;
15  import nl.mineleni.openls.databinding.gml.Pos;
16  
17  import org.custommonkey.xmlunit.XMLUnit;
18  import org.junit.Before;
19  import org.junit.Test;
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  import org.xml.sax.SAXException;
23  
24  /**
25   * Testcases voor {@link ReverseGeocodeRequest}.
26   * 
27   * @author mprins
28   */
29  public class ReverseGeocodeRequestTest extends AbstractTestUtils {
30  	private ReverseGeocodeRequest request;
31  	/** logger. */
32  	private static final Logger LOGGER = LoggerFactory
33  			.getLogger(ReverseGeocodeRequestTest.class);
34  
35  	/**
36  	 * set up xmlunit en testdata.
37  	 */
38  	@Before
39  	public void setUp() {
40  		XMLUnit.setIgnoreWhitespace(true);
41  		XMLUnit.setIgnoreComments(true);
42  
43  		Pos pos = new Pos();
44  		pos.setXY("7.986098 49.4434503");
45  		Point point = new Point();
46  		point.addPos(pos);
47  		point.setSrsName("EPSG:4326");
48  
49  		Position p = new Position();
50  		p.setPoint(point);
51  
52  		ReverseGeocodePreference pref = new ReverseGeocodePreference();
53  		pref.setPreference(ReverseGeocodePreference.PREFERENCE.StreetAddress
54  				.toString());
55  
56  		request = new ReverseGeocodeRequest();
57  		request.setPosition(p);
58  		request.setReverseGeocodePreference(pref);
59  	}
60  
61  	/**
62  	 * Testcase voor {@link ReverseGeocodeRequest#toXML()}.
63  	 * 
64  	 * @throws IOException
65  	 *             als er een fout optreed bij inlezen xml bestand
66  	 * @throws SAXException
67  	 *             als er een fout optreed bij parsen xml bestand
68  	 */
69  	@Test
70  	public void testToXML() throws IOException, SAXException {
71  		String control = readFileAsString("/samplerequests/reversegeocode/openls-reverse-geocode-request.xml");
72  		String test = request.toXML();
73  		LOGGER.debug("vergelijk: " + test + " met: " + control);
74  		assertXMLEqual(control, test);
75  	}
76  }
