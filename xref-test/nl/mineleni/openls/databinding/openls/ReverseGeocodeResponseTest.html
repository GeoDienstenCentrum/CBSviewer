8   
9   import static org.custommonkey.xmlunit.XMLAssert.assertXMLEqual;
10  
11  import java.io.IOException;
12  
13  import nl.mineleni.openls.AbstractTestUtils;
14  import nl.mineleni.openls.databinding.gml.Point;
15  import nl.mineleni.openls.databinding.gml.Pos;
16  
17  import org.custommonkey.xmlunit.XMLUnit;
18  import org.junit.Before;
19  import org.junit.Test;
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  import org.xml.sax.SAXException;
23  
24  /**
25   * Testcases voor {@link ReverseGeocodeResponse}.
26   * 
27   * @author mprins
28   */
29  public class ReverseGeocodeResponseTest extends AbstractTestUtils {
30  	private ReverseGeocodeResponse response;
31  	/** logger. */
32  	private static final Logger LOGGER = LoggerFactory
33  			.getLogger(ReverseGeocodeRequestTest.class);
34  
35  	/**
36  	 * set up testdata en xmlunit.
37  	 */
38  	@Before
39  	public void setUp() {
40  		XMLUnit.setIgnoreWhitespace(true);
41  		XMLUnit.setIgnoreComments(true);
42  
43  		Pos pos = new Pos();
44  		pos.setXY("7.986098 49.4434503");
45  		Point point = new Point();
46  		point.addPos(pos);
47  		point.setSrsName("EPSG:4326");
48  
49  		Street street = new Street();
50  		street.setStreet("Am Erlenbach");
51  		StreetAddress streetAddress = new StreetAddress();
52  		streetAddress.setStreet(street);
53  
54  		Place place = new Place();
55  		place.setPlace("Frankenstein");
56  		place.setType("Municipality");
57  
58  		Address address = new Address();
59  		address.setCountryCode("de");
60  		address.setStreetAddress(streetAddress);
61  		address.addPlace(place);
62  
63  		SearchCentreDistance searchCentreDistance = new SearchCentreDistance();
64  		searchCentreDistance.setUom("M");
65  		searchCentreDistance.setValue(115.6);
66  
67  		ReverseGeocodedLocation reverseGeocodedLocation = new ReverseGeocodedLocation();
68  		reverseGeocodedLocation.setAddress(address);
69  		reverseGeocodedLocation.setSearchCentreDistance(searchCentreDistance);
70  		reverseGeocodedLocation.setPoint(point);
71  
72  		response = new ReverseGeocodeResponse();
73  		response.setReverseGeocodedLocation(reverseGeocodedLocation);
74  	}
75  
76  	/**
77  	 * Testcase voor {@link ReverseGeocodeResponse#toXML()}.
78  	 * 
79  	 * @throws IOException
80  	 *             als er een fout optreed bij inlezen xml bestand
81  	 * @throws SAXException
82  	 *             als er een fout optreed bij parsen xml bestand
83  	 */
84  	@Test
85  	public void testToXML() throws IOException, SAXException {
86  		String control = readFileAsString("/sampleresponses/reversegeocode/response.xml");
87  		String test = response.toXML();
88  		LOGGER.debug("vergelijk: " + test + " met: " + control);
89  		assertXMLEqual(control, test);
90  	}
91  }
