  7 /**
  8  * Functies voor het zoekformulier zoals gedefinieerd in zoekformulier.jsp.
  9  * 
 10  * @fileoverview zorgt voor afhandeling van het adreszoek formulier.
 11  * @author mprins
 12  * @requires jQuery
 13  * @requires jQueryUI
 14  * @requires zoekformulier.jsp
 15  * @requires Viewer.js
 16  * @class {ZoekFormulier}
 17  */
 18 var ZoekFormulier = {
 19 	/**
 20 	 * Aanhaken van autocomplete handler aan input veld van zoek formulier.
 21 	 * 
 22 	 * @constructor
 23 	 */
 24 	init : function(viewer) {
 25 		// de zoekknop, method en action van formulier verwijderen, die hebben geen funtie meer in de RIA
 26 		jQuery('input:submit').attr("disabled", true);
 27 		jQuery('#zoekFormulier').removeAttr('action');
 28 		jQuery('#zoekFormulier').removeAttr('method');
 29 
 30 		// verborgen velden instellen voor RIA functies
 31 		jQuery('#zoekFormulier').find('input[name="coreonly"]').val('false');
 32 		jQuery('#zoekFormulier').find('input[name="forward"]').val('false');
 33 
 34 		// aanhaken auto complete
 35 		jQuery('#adres').autocomplete({
 36 			source : function(request, response) {
 37 				jQuery.ajax({
 38 					url : 'adres',
 39 					data : {
 40 						adres : request.term,
 41 						forward : false,
 42 						format : 'json',
 43 						coreonly : false
 44 					},
 45 					dataType : 'json'
 46 				}).success(function(data) {
 47 					var results = jQuery.map(data, function(val, i) {
 48 						return {
 49 							label : val.addressString,
 50 							x : val.xCoord,
 51 							y : val.yCoord,
 52 							r : val.radius
 53 						};
 54 					});
 55 					response(results);
 56 				});
 57 			},
 58 			select : function(event, ui) {
 59 				// console.debug(ui.item ? "Gekozen: " + ui.item.value + "
 60 				// (x,y,r) :" + ui.item.x + ","
 61 				// + ui.item.y + "," + ui.item.r : "Niets gekozen, de input was
 62 				// " + this.value);
 63 				if (ui.item) {
 64 					Viewer.zoomTo(ui.item.x, ui.item.y, ui.item.r, true);
 65 					//Viewer.featureInfo(ui.item.x, ui.item.y);
 66 				}
 67 			},
 68 			// alleen items uit de lijst mogen ingevuld/gekozen worden
 69 			change : function(event, ui) {
 70 				if (!ui.item) {
 71 					jQuery(this).val('');
 72 				}
 73 			},
 74 			minLength : 4 /* characters */,
 75 			delay : 400 /* milliseconds */,
 76 			autoFocus : false
 77 		});
 78 	}
 79 };
