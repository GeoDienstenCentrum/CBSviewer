  7 /**
  8  * Functies voor het zoekformulier zoals gedefinieerd in zoekformulier.jsp.
  9  * 
 10  * @fileoverview zorgt voor afhandeling van het adreszoek formulier.
 11  * @author mprins
 12  * @requires jQuery
 13  * @requires zoekformulier.jsp
 14  * @requires Viewer.js
 15  * @class
 16  */
 17 var ZoekFormulier = {
 18 	/**
 19 	 * aanhaken van handler aan submit event van formulier.
 20 	 * 
 21 	 * @constructor
 22 	 */
 23 	init : function() {
 24 		jQuery('#zoekFormulier').ajaxForm({
 25 			beforeSerialize : this.handleBeforeSerialize,
 26 			beforeSubmit: this.handleBeforeSubmit,
 27 			success : this.handleSuccess,
 28 			dataType : 'json',
 29 			data : {
 30 				format : 'json'
 31 			}
 32 		});
 33 	},
 34 	/**
 35 	 * Bijwerken van verborgen velden in het formulier.
 36 	 * 
 37 	 * @param {String}
 38 	 *            $form html snippet
 39 	 * @param {Object}
 40 	 *            options een jQuery.ajax options object
 41 	 * @returns {Boolean} true to continue the request
 42 	 * 
 43 	 * @see http://docs.jquery.com/Ajax/jQuery.ajax#options
 44 	 * @private
 45 	 */
 46 	handleBeforeSerialize : function($form, options) {
 47 		$form.find('input[name="coreonly"]').val('false');
 48 		$form.find('input[name="forward"]').val('false');
 49 		return true;
 50 	},
 51 	/**
 52 	 * Controleren of het formulier juist is ingevuld.
 53 	 * 
 54 	 * @param {String}
 55 	 *            $form html snippet
 56 	 * @param {Object}
 57 	 *            options een jQuery.ajax options object
 58 	 * @returns {Boolean} true to continue the request
 59 	 * 
 60 	 * @see http://docs.jquery.com/Ajax/jQuery.ajax#options
 61 	 * @private
 62 	 */
 63 	handleBeforeSubmit : function($form, options) {
 64 		if (jQuery('#adres[title]').val() === jQuery('#adres').attr('title'))
 65 			return false;
 66 			
 67 		return true;
 68 	},
 69 	/**
 70 	 * Actief maken van de geselecteerde zoekknop
 71 	 * 
 72 	 * @param {id}
 73 	 *            button id
 74 	 * 
 75 	 * @private
 76 	 */
 77 	toggleSelectedButton : function(id) {
 78 		jQuery('.adreslijst').find('li a').removeClass("selected");
 79 		jQuery("#button" + id).addClass('selected');
 80 	},
 81 	
 82 	/**
 83 	 * Tonen van een placeholder in het zoek veld
 84 	 * Dit kan vanaf html5 ook met het placeholder attribuut alleen deze komt niet door de W3C check
 85 	 *
 86 	 * @param (props)
 87 	 *
 88 	 * @private
 89 	 */
 90 	addPlaceholders: function(props) {
 91 		//jQuery.extend($.support, {
 92 		//	placeHolder: !!(jQuery('#adres', zoekFormulier)[0].placeholder !== undefined || jQuery('#adres', zoekFormulier)[0].placeHolder !== undefined)
 93 		//});
 94 
 95 		// create placeholders by using "value" attribute
 96 		//if (!jQuery.support.placeHolder) {
 97 	
 98 			var placeholder = jQuery('#adres').attr('title');
 99 			jQuery('#adres[title]').bind({
100 				'focus': function() {
101 					if (jQuery(this).val() === placeholder) {
102 						jQuery(this).val('');
103 						jQuery(this).removeClass('adresEmpty');
104 					}
105 					jQuery(this).attr('data-focused', 'yes');
106 				},
107 
108 				'blur': function() {
109 					if (jQuery(this).val() === '') {
110 						jQuery(this).val(placeholder);
111 						jQuery(this).addClass('adresEmpty');
112 					}
113 					jQuery(this).removeAttr('data-focused');
114 				}
115 			});
116 
117 			// only add placeholder on load when value is empty or placeholder or input is not focused (focus is preserved while reloading/XHR)
118 			if (((jQuery('#adres').val() === '') || (jQuery('#adres').val() === placeholder)) && (!jQuery('#adres').attr('data-focused'))) {
119 				jQuery('#adres').val(placeholder);
120 				jQuery('#adres').addClass('adresEmpty');
121 			}
122 		//}
123 	},
124 	
125 	/**
126 	 * Bijwerken van de pagina en eventueel de kaart inzoomen.
127 	 * 
128 	 * @param {Array}
129 	 *            data json array met adreslijst
130 	 * @param {String}
131 	 *            statusText response status tekst
132 	 * @param xhr
133 	 *            jQuery XHR
134 	 * @param {String}
135 	 *            $form html snippet
136 	 * 
137 	 * @private
138 	 */
139 	handleSuccess : function(data, statusText, xhr, $form) {
140 		var html = "";
141 		var paragraaf = jQuery('#zoekresultaten');
142 
143 		// in het geval dat een eerdere actie niets opleverde
144 		paragraaf.removeClass('error');
145 
146 		switch (data.length) {
147 		case 0:
148 			paragraaf.addClass('error');
149 			html += OpenLayers.i18n("KEY_ZOEKEN_GEEN_ADRES");
150 			break;
151 		case 1:
152 			html += OpenLayers.i18n("KEY_ZOEKEN_EEN_ADRES");
153 			html += '<ul class="adreslijst"><li>' + data[0].addressString + '</li></ul>';
154 			Viewer.zoomTo(data[0].xCoord, data[0].yCoord, data[0].radius);
155 			break;
156 		default:
157 			// html += '<p style="clear:both">' +
158 			// OpenLayers.i18n("KEY_ZOEKEN_MEER_ADRES") + '</p>';
159 			html += '<ul class="adreslijst">';
160 
161 			for ( var i = 0; i < data.length; i++) {
162 				var selected = "";
163 				if (i === 0) {
164 					selected = " selected";
165 				}
166 				html += '<li><a id="button' + i + '" class="button' + selected
167 						+ '" href="#" onclick="ZoekFormulier.toggleSelectedButton(' + i + ');Viewer.zoomTo('
168 						+ data[i].xCoord + ',' + data[i].yCoord + ',' + data[i].radius + ');return false;" title="'
169 						+ OpenLayers.i18n("KEY_ZOEKEN_LINK_TTL", {
170 							'0' : '' + data[i].addressString
171 						}) + '">' + data[i].addressString + '</a></li>';
172 			}
173 			html += '</ul>';
174 
175 			Viewer.zoomTo(data[0].xCoord, data[0].yCoord, data[0].radius);
176 		}
177 		paragraaf.empty().html(html);
178 	}
179 };
