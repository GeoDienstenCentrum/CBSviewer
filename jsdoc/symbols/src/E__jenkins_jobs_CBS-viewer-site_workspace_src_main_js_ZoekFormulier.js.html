1 /**
  2  * Functies voor het zoekformulier zoals gedefinieerd in zoekformulier.jsp.
  3  * 
  4  * @fileoverview zorgt voor afhandeling van het adreszoek formulier.
  5  * @author mprins
  6  * @requires jQuery
  7  * @requires zoekformulier.jsp
  8  * @requires Viewer.js
  9  * @class
 10  */
 11 var ZoekFormulier = {
 12 	/**
 13 	 * aanhaken van handler aan submit event van formulier.
 14 	 * 
 15 	 * @constructor
 16 	 */
 17 	init : function() {
 18 		jQuery('#zoekFormulier').ajaxForm({
 19 			beforeSerialize : this.handleBeforeSerialize,
 20 			success : this.handleSuccess,
 21 			dataType : 'json',
 22 			data : {
 23 				format : 'json'
 24 			}
 25 		});
 26 	},
 27 	/**
 28 	 * Bijwerken van verborgen velden in het formulier.
 29 	 * 
 30 	 * @param {String}
 31 	 *            $form html snippet
 32 	 * @param {Object}
 33 	 *            options een jQuery.ajax options object
 34 	 * @returns {Boolean} true to continue the request
 35 	 * 
 36 	 * @see http://docs.jquery.com/Ajax/jQuery.ajax#options
 37 	 * @private
 38 	 */
 39 	handleBeforeSerialize : function($form, options) {
 40 		$form.find('input[name="coreonly"]').val('false');
 41 		$form.find('input[name="forward"]').val('false');
 42 		return true;
 43 	},
 44 	/**
 45 	 * Bijwerken van de pagina en eventueel de kaart inzoomen.
 46 	 * 
 47 	 * @param {Array}
 48 	 *            data json array met adreslijst
 49 	 * @param {String}
 50 	 *            statusText response status tekst
 51 	 * @param xhr
 52 	 *            jQuery XHR
 53 	 * @param {String}
 54 	 *            $form html snippet
 55 	 * 
 56 	 * @todo Strings van de meldingen overbrengen naar resource bundle
 57 	 * 
 58 	 * @private
 59 	 */
 60 	handleSuccess : function(data, statusText, xhr, $form) {
 61 		var html = "";
 62 		var paragraaf = jQuery('#zoekresultaten');
 63 
 64 		// in het geval dat een eerdere actie niets opleverde
 65 		paragraaf.removeClass('error');
 66 
 67 		switch (data.length) {
 68 		case 0:
 69 			paragraaf.addClass('error');
 70 			html += 'Er is geen adres gevonden.';
 71 			break;
 72 		case 1:
 73 			html += 'Er is een adres gevonden.';
 74 			html += '<ul class="adreslijst"><li>' + data[0].addressString + '</li></ul>';
 75 			Viewer.zoomTo(data[0].xCoord, data[0].yCoord, data[0].radius);
 76 			break;
 77 		default:
 78 			html += 'Er zijn ' + data.length + ' adressen gevonden.';
 79 			html += '<ul class="adreslijst">';
 80 			for ( var i = 0; i < data.length; i++) {
 81 				html += '<li>' + '<a href="#" onclick="Viewer.zoomTo(' + data[i].xCoord + ',' + data[i].yCoord + ','
 82 						+ data[i].radius + ');return false;" title="Zoom naar adres">' + data[i].addressString
 83 						+ '</a></li>';
 84 			}
 85 			html += '</ul>';
 86 		}
 87 		paragraaf.empty().html(html);
 88 	},
 89 };
