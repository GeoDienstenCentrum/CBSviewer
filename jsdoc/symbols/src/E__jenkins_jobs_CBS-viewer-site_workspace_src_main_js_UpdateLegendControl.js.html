  7 /**
  8  * Werkt de legenda bij als de kaart wordt ge-update met een WMS layer.
  9  * 
 10  * @author mprins
 11  * @class
 12  * @extends OpenLayers.Control
 13  * @requires OpenLayers/Control.js
 14  * @requires OpenLayers/Layer/WMS/js
 15  * @constructor
 16  */
 17 UpdateLegendControl = OpenLayers.Class(OpenLayers.Control, {
 18 	/**
 19 	 * Activate the control when it is added to a map. Default is true.
 20 	 * 
 21 	 * @type{Boolean}
 22 	 */
 23 	autoActivate : true,
 24 
 25 	/**
 26 	 * Dom element waar de legenda in wordt gevoegd. Moet een container element
 27 	 * zijn zoals P, DIV, SPAN, MAP, LI, ...
 28 	 * 
 29 	 * @type {DOMElement}
 30 	 */
 31 	element : null,
 32 
 33 	/**
 34 	 * destroy this control.
 35 	 */
 36 	destroy : function() {
 37 		this.deactivate();
 38 		OpenLayers.Control.prototype.destroy.apply(this, arguments);
 39 	},
 40 
 41 	/**
 42 	 * activate this control.
 43 	 */
 44 	activate : function() {
 45 		if (OpenLayers.Control.prototype.activate.apply(this, arguments)) {
 46 			this.map.events.register('addlayer', this, this.redraw);
 47 			this.redraw();
 48 			return true;
 49 		} else {
 50 			return false;
 51 		}
 52 	},
 53 
 54 	/**
 55 	 * deactivate this control.
 56 	 */
 57 	deactivate : function() {
 58 		if (OpenLayers.Control.prototype.deactivate.apply(this, arguments)) {
 59 			this.map.events.unregister('addlayer', this, this.redraw);
 60 			this.element.innerHTML = "";
 61 			return true;
 62 		} else {
 63 			return false;
 64 		}
 65 	},
 66 
 67 	/**
 68 	 * draw this control.
 69 	 * 
 70 	 * @returns {DOMElement}
 71 	 */
 72 	draw : function() {
 73 		OpenLayers.Control.prototype.draw.apply(this, arguments);
 74 		if (!this.element) {
 75 			this.div.left = "";
 76 			this.div.top = "";
 77 			this.element = this.div;
 78 		}
 79 		return this.div;
 80 	},
 81 
 82 	/**
 83 	 * redraw this control.
 84 	 * 
 85 	 * @param {OpenLayers.Event}
 86 	 *            evt De addlayer event
 87 	 */
 88 	redraw : function(evt) {
 89 		// addlayer event - triggered after a layer has been added. The event
 90 		// object will include a *layer* property that references the added
 91 		// layer.
 92 		if (evt == null) {
 93 			this.reset();
 94 			return;
 95 		} else {
 96 			// test if layer is WMS
 97 			if (evt.layer instanceof OpenLayers.Layer.WMS) {
 98 				var lyr = evt.layer;
 99 				var legendUrl = lyr.url + '?SERVICE=WMS&REQUEST=GetLegendGraphic' + '&FORMAT=image%2Fpng'
100 						+ '&WIDTH=25&HEIGHT=25' + '&LAYER=' + lyr.params.LAYERS + '&STYLE=' + lyr.params.STYLES
101 						+ '&VERSION=' + lyr.params.VERSION + '&EXCEPTIONS=' + lyr.params.EXCEPTIONS;
102 				var newHtml = '<img src="' + legendUrl + '" alt="Legenda voor kaartlaag ' + lyr.name + '"/>';
103 
104 				if (newHtml != this.element.innerHTML) {
105 					this.element.innerHTML = newHtml;
106 				}
107 			}
108 		}
109 	},
110 
111 	/**
112 	 * reset this control.
113 	 * 
114 	 * @param {OpenLayers.Event}
115 	 *            evt De addlayer event
116 	 */
117 	reset : function(evt) {
118 		if (this.emptyString !== null) {
119 			this.element.innerHTML = this.emptyString;
120 		}
121 	}
122 });
