  7 /**
  8  * Een control die een ikoontje tekent op de plaats waar geklikt is.
  9  * 
 10  * @author mprins
 11  * @class {ClickDrawControl}
 12  * @extends {OpenLayers.Control}
 13  */
 14 ClickDrawControl = OpenLayers.Class(OpenLayers.Control, {
 15 	/** @override */
 16 	autoActivate : true,
 17 
 18 	/** @override */
 19 	defaultHandlerOptions : {
 20 		'single' : true,
 21 		'double' : false,
 22 		'pixelTolerance' : 0,
 23 		'stopSingle' : false,
 24 		'stopDouble' : false
 25 	},
 26 
 27 	/** @constructor */
 28 	initialize : function(options) {
 29 		this.handlerOptions = OpenLayers.Util.extend({}, this.defaultHandlerOptions);
 30 		OpenLayers.Control.prototype.initialize.apply(this, arguments);
 31 		this.handler = new OpenLayers.Handler.Click(this, {
 32 			'click' : this.trigger
 33 		}, this.handlerOptions);
 34 	},
 35 
 36 	/**
 37 	 * Klik event handler.
 38 	 * 
 39 	 * @param e
 40 	 *            {OpenLayers.Event} een muisklik event
 41 	 */
 42 	trigger : function(e) {
 43 		if (!this.layer) {
 44 			this.makeLayer();
 45 		}
 46 		this.drawOrMove(e);
 47 	},
 48 
 49 	/**
 50 	 * zorgt voor een vector layer om een marker in te tekenen.
 51 	 * 
 52 	 * @private
 53 	 */
 54 	makeLayer : function() {
 55 		this.layer = new OpenLayers.Layer.Vector(this.CLASS_NAME, {
 56 			styleMap : new OpenLayers.StyleMap({
 57 				externalGraphic : 'img/info.png',
 58 				graphicHeight : 37,
 59 				graphicWidth : 32,
 60 				graphicXOffset : -16,
 61 				graphicYOffset : -37
 62 			})
 63 		});
 64 		this.map.addLayer(this.layer);
 65 	},
 66 
 67 	/**
 68 	 * teken een ikoontje.
 69 	 * 
 70 	 * @param e
 71 	 *            {OpenLayers.Event}
 72 	 * @private
 73 	 */
 74 	drawOrMove : function(e) {
 75 		// verplaats naar bovenste
 76 		this.map.setLayerIndex(this.layer, this.map.getNumLayers()-1);
 77 		var lonlat = this.map.getLonLatFromPixel(e.xy);
 78 		if (!this.point) {
 79 			var geometry = new OpenLayers.Geometry.Point(lonlat.lon, lonlat.lat);
 80 			this.point = new OpenLayers.Feature.Vector(geometry);
 81 			this.point.geometry.clearBounds();
 82 			this.layer.addFeatures([ this.point ], {
 83 				silent : true
 84 			});
 85 		} else {
 86 			this.point.geometry.x = lonlat.lon;
 87 			this.point.geometry.y = lonlat.lat;
 88 			this.point.geometry.clearBounds();
 89 		}
 90 		this.layer.drawFeature(this.point);
 91 	},
 92 
 93 	/**
 94 	 * activate deze control.
 95 	 * 
 96 	 * @returns {Boolean}
 97 	 * @override
 98 	 */
 99 	activate : function() {
100 		if (!OpenLayers.Control.prototype.activate.apply(this, arguments)) {
101 			return false;
102 		}
103 		return true;
104 	},
105 
106 	/**
107 	 * deactivate deze control, zorgt oa. voor opruimen van resources.
108 	 * 
109 	 * @returns {Boolean}
110 	 * @override
111 	 */
112 	deactivate : function() {
113 		if (!OpenLayers.Control.prototype.deactivate.apply(this, arguments)) {
114 			return false;
115 		}
116 		if (this.layer) {
117 			this.map.removeLayer(this.layer);
118 			this.layer.destroyFeatures([ this.point ]);
119 			this.point = null;
120 			this.layer = null;
121 		}
122 		return true;
123 	},
124         
125 	CLASS_NAME : 'ClickDrawControl'
126 });
