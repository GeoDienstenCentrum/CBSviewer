1 /**
  2  * Viewer klasse.
  3  * 
  4  * @author mprins
  5  */
  6 Viewer = function() {
  7 	/**
  8 	 * 
  9 	 * @private
 10 	 * @type OpenLayers.Map
 11 	 */
 12 	var _map;
 13 
 14 	return {
 15 		/**
 16 		 * Constructor.
 17 		 * 
 18 		 * @param config
 19 		 *            Configratie object
 20 		 * @constructor
 21 		 */
 22 		init : function(config) {
 23 			this.config = config;
 24 			// reset any controls
 25 			jQuery.extend(true, this.config, {
 26 				map : {
 27 					controls : []
 28 				}
 29 			});
 30 			jQuery(window).unload(function() {
 31 				Viewer.destroy();
 32 			});
 33 			OpenLayers.Lang.setCode('nl');
 34 			OpenLayers.Number.decimalSeparator = ",";
 35 
 36 			this.createMap();
 37 			this.addBaseMap();
 38 			this.addControls();
 39 		},
 40 
 41 		/**
 42 		 * Initialiseer de kaart.
 43 		 */
 44 		createMap : function() {
 45 			_map = new OpenLayers.Map(this.config.mapDiv, this.config.map);
 46 		},
 47 
 48 		/**
 49 		 * 
 50 		 * @returns OpenLayers.Map object van deze Viewer of undefined als het
 51 		 *          object niet is geinitialiseerd
 52 		 * @deprecated probeer deze niet te gebruiken
 53 		 */
 54 		getMap : function() {
 55 			return _map;
 56 		},
 57 
 58 		/**
 59 		 * add the controls
 60 		 */
 61 		addControls : function() {
 62 			_map.addControl(new OpenLayers.Control.KeyboardDefaults({
 63 				/* alleen actief als de kaart focus heeft */
 64 				observeElement : this.config.mapDiv
 65 			}));
 66 			_map.addControl(new OpenLayers.Control.Zoom());
 67 			_map.addControl(new OpenLayers.Control.Navigation({
 68 				zoomWheelEnabled : false
 69 			}));
 70 		},
 71 
 72 		/**
 73 		 * cleanup.
 74 		 */
 75 		destroy : function() {
 76 			_map = null;
 77 		},
 78 
 79 		/**
 80 		 * add the WMS to the map.
 81 		 */
 82 		addWMS : function(wmsConfig) {
 83 			var layer = new OpenLayers.Layer.WMS(wmsConfig.name, wmsConfig.url, {
 84 				layers : wmsConfig.layers,
 85 				styles : wmsConfig.styles,
 86 				version : '1.1.1',
 87 				format : 'image/png',
 88 				transparent : true
 89 			}, {
 90 				isBaseLayer : false,
 91 				visibility : true,
 92 				singleTile : true
 93 			});
 94 			_map.addLayer(layer);
 95 		},
 96 
 97 		/**
 98 		 * removes the WMS from the map.
 99 		 * 
100 		 * @param wmsID
101 		 *            ID van de WMS
102 		 */
103 		removeWMS : function(wmsID) {
104 			// TODO implementatie
105 		},
106 
107 		/**
108 		 * set up basemap.
109 		 */
110 		addBaseMap : function() {
111 			var matrixIds = [ 13 ];
112 			for ( var i = 0; i < 13; ++i) {
113 				matrixIds[i] = "EPSG:28992:" + i;
114 			}
115 			var lyr = new OpenLayers.Layer.WMTS({
116 				name : "achtergrond",
117 				url : "http://geodata.nationaalgeoregister.nl/tiles/service/wmts/brtachtergrondkaart/",
118 				layer : "brtachtergrondkaart",
119 				matrixSet : 'EPSG:28992',
120 				matrixIds : matrixIds,
121 				format : 'image/png8',
122 				style : '_null'
123 			});
124 			_map.addLayer(lyr);
125 			_map.zoomTo(6);
126 		}
127 	};
128 }();
