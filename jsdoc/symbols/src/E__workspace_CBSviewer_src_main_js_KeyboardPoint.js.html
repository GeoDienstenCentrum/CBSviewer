 11 /**
 12  * Class: OpenLayers.Handler.KeyboardPoint
 13  * 
 14  * A custom handler that displays a vector point that can be moved using the
 15  * arrow keys of the keyboard.
 16  */
 17 OpenLayers.Handler.KeyboardPoint = OpenLayers.Class(OpenLayers.Handler, {
 18 
 19 	KEY_EVENTS : [ "keydown" ],
 20 
 21 	initialize : function(control, callbacks, options) {
 22 		OpenLayers.Handler.prototype.initialize.apply(this, arguments);
 23 		// cache the bound event listener method so it can be unobserved later
 24 		this.eventListener = OpenLayers.Function.bindAsEventListener(
 25 				this.handleKeyEvent, this);
 26 	},
 27 
 28 	activate : function() {
 29 		if (!OpenLayers.Handler.prototype.activate.apply(this, arguments)) {
 30 			return false;
 31 		}
 32 		this.layer = new OpenLayers.Layer.Vector(this.CLASS_NAME);
 33 		this.map.addLayer(this.layer);
 34 		this.observeElement = this.observeElement || document;
 35 		for ( var i = 0, len = this.KEY_EVENTS.length; i < len; i++) {
 36 			OpenLayers.Event.observe(this.observeElement, this.KEY_EVENTS[i],
 37 					this.eventListener);
 38 		}
 39 		if (!this.point) {
 40 			this.createFeature();
 41 		}
 42 		return true;
 43 	},
 44 
 45 	deactivate : function() {
 46 		if (!OpenLayers.Handler.prototype.deactivate.apply(this, arguments)) {
 47 			return false;
 48 		}
 49 		for ( var i = 0, len = this.KEY_EVENTS.length; i < len; i++) {
 50 			OpenLayers.Event.stopObserving(this.observeElement,
 51 					this.KEY_EVENTS[i], this.eventListener);
 52 		}
 53 		this.map.removeLayer(this.layer);
 54 		this.destroyFeature();
 55 		return true;
 56 	},
 57 
 58 	handleKeyEvent : function(evt) {
 59 		switch (evt.keyCode) {
 60 		case OpenLayers.Event.KEY_LEFT:
 61 			this.modifyFeature(-3, 0);
 62 			break;
 63 		case OpenLayers.Event.KEY_RIGHT:
 64 			this.modifyFeature(3, 0);
 65 			break;
 66 		case OpenLayers.Event.KEY_UP:
 67 			this.modifyFeature(0, 3);
 68 			break;
 69 		case OpenLayers.Event.KEY_DOWN:
 70 			this.modifyFeature(0, -3);
 71 			break;
 72 		case OpenLayers.Event.KEY_RETURN:
 73 			this.callback('done', [ this.point.geometry.clone() ]);
 74 			break;
 75 		case OpenLayers.Event.KEY_ESC:
 76 			this.callback('cancel');
 77 			break;
 78 		}
 79 	},
 80 
 81 	modifyFeature : function(lon, lat) {
 82 		if (!this.point) {
 83 			this.createFeature();
 84 		}
 85 		var resolution = this.map.getResolution();
 86 		this.point.geometry.x = this.point.geometry.x + lon * resolution;
 87 		this.point.geometry.y = this.point.geometry.y + lat * resolution;
 88 		this.callback("modify", [ this.point.geometry, this.point, false ]);
 89 		this.point.geometry.clearBounds();
 90 		this.drawFeature();
 91 	},
 92 
 93 	createFeature : function() {
 94 		var center = this.map.getCenter();
 95 		var geometry = new OpenLayers.Geometry.Point(center.lon, center.lat);
 96 		this.point = new OpenLayers.Feature.Vector(geometry);
 97 		this.callback("create", [ this.point.geometry, this.point ]);
 98 		this.point.geometry.clearBounds();
 99 		this.layer.addFeatures([ this.point ], {
100 			silent : true
101 		});
102 	},
103 
104 	destroyFeature : function() {
105 		this.layer.destroyFeatures([ this.point ]);
106 		this.point = null;
107 	},
108 
109 	drawFeature : function() {
110 		this.layer.drawFeature(this.point, this.style);
111 	}
112 });
