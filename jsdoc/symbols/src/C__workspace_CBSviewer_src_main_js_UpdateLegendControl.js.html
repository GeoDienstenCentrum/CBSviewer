1 /**
  2  * Werkt de legenda bij als de kaart wordt ge-update met een WMS layer.
  3  * 
  4  * @author mprins
  5  * @class
  6  * @extends OpenLayers.Control
  7  * @requires OpenLayers/Control.js
  8  * @requires OpenLayers/Layer/WMS/js
  9  * @constructor
 10  */
 11 UpdateLegendControl = OpenLayers.Class(OpenLayers.Control, {
 12 	/**
 13 	 * Activate the control when it is added to a map. Default is true.
 14 	 * 
 15 	 * @type{Boolean}
 16 	 */
 17 	autoActivate : true,
 18 
 19 	/**
 20 	 * Dom element waar de legenda in wordt gevoegd. Moet een container element
 21 	 * zijn zoals P, DIV, SPAN, MAP, LI, ...
 22 	 * 
 23 	 * @type {DOMElement}
 24 	 */
 25 	element : null,
 26 
 27 	/**
 28 	 * destroy this control.
 29 	 */
 30 	destroy : function() {
 31 		this.deactivate();
 32 		OpenLayers.Control.prototype.destroy.apply(this, arguments);
 33 	},
 34 
 35 	/**
 36 	 * activate this control.
 37 	 */
 38 	activate : function() {
 39 		if (OpenLayers.Control.prototype.activate.apply(this, arguments)) {
 40 			this.map.events.register('addlayer', this, this.redraw);
 41 			this.redraw();
 42 			return true;
 43 		} else {
 44 			return false;
 45 		}
 46 	},
 47 
 48 	/**
 49 	 * deactivate this control.
 50 	 */
 51 	deactivate : function() {
 52 		if (OpenLayers.Control.prototype.deactivate.apply(this, arguments)) {
 53 			this.map.events.unregister('addlayer', this, this.redraw);
 54 			this.element.innerHTML = "";
 55 			return true;
 56 		} else {
 57 			return false;
 58 		}
 59 	},
 60 
 61 	/**
 62 	 * draw this control.
 63 	 * 
 64 	 * @returns {DOMElement}
 65 	 */
 66 	draw : function() {
 67 		OpenLayers.Control.prototype.draw.apply(this, arguments);
 68 		if (!this.element) {
 69 			this.div.left = "";
 70 			this.div.top = "";
 71 			this.element = this.div;
 72 		}
 73 		return this.div;
 74 	},
 75 
 76 	/**
 77 	 * redraw this control.
 78 	 * 
 79 	 * @param {OpenLayers.Event}
 80 	 *            evt De addlayer event
 81 	 */
 82 	redraw : function(evt) {
 83 		// addlayer event - triggered after a layer has been added. The event
 84 		// object will include a *layer* property that references the added
 85 		// layer.
 86 		if (evt == null) {
 87 			this.reset();
 88 			return;
 89 		} else {
 90 			// test if layer is WMS
 91 			if (evt.layer instanceof OpenLayers.Layer.WMS) {
 92 				var lyr = evt.layer;
 93 				var legendUrl = lyr.url + '?SERVICE=WMS&REQUEST=GetLegendGraphic' + '&FORMAT=image%2Fpng'
 94 						+ '&WIDTH=20&HEIGHT=20' + '&LAYER=' + lyr.params.LAYERS + '&STYLES=' + lyr.params.STYLES
 95 						+ '&VERSION=' + lyr.params.VERSION + '&EXCEPTIONS=' + lyr.params.EXCEPTIONS;
 96 				var newHtml = '<img src="' + legendUrl + '" alt="Legenda voor kaartlaag ' + lyr.name + '"/>';
 97 
 98 				if (newHtml != this.element.innerHTML) {
 99 					this.element.innerHTML = newHtml;
100 				}
101 			}
102 		}
103 	},
104 
105 	/**
106 	 * reset this control.
107 	 * 
108 	 * @param {OpenLayers.Event}
109 	 *            evt De addlayer event
110 	 */
111 	reset : function(evt) {
112 		if (this.emptyString !== null) {
113 			this.element.innerHTML = this.emptyString;
114 		}
115 	}
116 });
