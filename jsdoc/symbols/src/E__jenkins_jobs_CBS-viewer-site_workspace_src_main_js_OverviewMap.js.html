  7 /**
  8  * Overzichtskaart, gebaseerd op OpenLayers.Control.OverviewMap.
  9  * 
 10  * @author mprins
 11  * @class {OverviewMap}
 12  * @extends {OpenLayers.Control.OverviewMap}
 13  */
 14 OverviewMap = OpenLayers.Class(OpenLayers.Control.OverviewMap, {
 15 
 16 	/** @override */
 17 	maximized : true,
 18 
 19 	/** @override */
 20 	mapOptions : {
 21 		theme : null
 22 	},
 23 
 24 	/** @override */
 25 	size : {
 26 		w : 140,
 27 		h : 140
 28 	},
 29 
 30 	/** @override */
 31 	autoPan : true,
 32 
 33 	/**
 34 	 * Constructor: OpenLayers.Control.OverviewMap Create a new overview map
 35 	 * 
 36 	 * Parameters: options - {Object} Properties of this object will be set on
 37 	 * the overview map object. Note, to set options on the map object contained
 38 	 * in this control, set <mapOptions> as one of the options properties.
 39 	 * 
 40 	 * @override
 41 	 */
 42 	initialize : function(options) {
 43 		this.layers = [];
 44 		this.handlers = {};
 45 		OpenLayers.Control.prototype.initialize.apply(this, [ options ]);
 46 	},
 47 
 48 	/**
 49 	 * Method: draw Render the control in the browser.
 50 	 * 
 51 	 * @override
 52 	 */
 53 	draw : function() {
 54 		OpenLayers.Control.prototype.draw.apply(this, arguments);
 55 		if (this.layers.length === 0) {
 56 			if (this.map.baseLayer) {
 57 				var layer = this.map.baseLayer.clone();
 58 				this.layers = [ layer ];
 59 			} else {
 60 				this.map.events.register("changebaselayer", this, this.baseLayerDraw);
 61 				return this.div;
 62 			}
 63 		}
 64 
 65 		// create overview map DOM elements
 66 		this.element = document.createElement('div');
 67 		this.element.className = this.displayClass + 'Element';
 68 		this.element.style.display = 'none';
 69 
 70 		this.mapDiv = document.createElement('div');
 71 		this.mapDiv.style.width = this.size.w + 'px';
 72 		this.mapDiv.style.height = this.size.h + 'px';
 73 		this.mapDiv.style.position = 'relative';
 74 		this.mapDiv.style.overflow = 'hidden';
 75 		this.mapDiv.id = OpenLayers.Util.createUniqueID('overviewMap');
 76 
 77 		this.extentRectangle = document.createElement('div');
 78 		this.extentRectangle.style.position = 'absolute';
 79 		this.extentRectangle.style.zIndex = 1000; // HACK
 80 		this.extentRectangle.className = this.displayClass + 'ExtentRectangle';
 81 
 82 		this.element.appendChild(this.mapDiv);
 83 
 84 		this.div.appendChild(this.element);
 85 
 86 		this.div.className += " " + this.displayClass + 'Container';
 87 
 88 		// maximize button
 89 		var btn = document.createElement("button");
 90 		btn.innerHTML = '+';
 91 		// btn.type = 'button';
 92 		btn.name = 'tonen';
 93 		this.maximizeDiv = btn;
 94 		this.maximizeDiv.style.display = 'none';
 95 		this.maximizeDiv.className = this.displayClass + 'MaximizeButton overviewMapButton olButton';
 96 		this.maximizeDiv.title = OpenLayers.i18n('KEY_TOGGLE_OVERVIEW_ON');
 97 		this.div.appendChild(this.maximizeDiv);
 98 
 99 		// minimize button
100 		btn = document.createElement("button");
101 		btn.innerHTML = '−';
102 		// btn.type = 'button';
103 		btn.name = 'verbergen';
104 		this.minimizeDiv = btn;
105 		this.minimizeDiv.style.display = 'none';
106 		this.minimizeDiv.className = this.displayClass + 'MinimizeButton overviewMapButton olButton';
107 		this.minimizeDiv.title = OpenLayers.i18n('KEY_TOGGLE_OVERVIEW_OFF');
108 		this.div.appendChild(this.minimizeDiv);
109 
110 		this.minimizeControl();
111 
112 		if (this.map.getExtent()) {
113 			this.update();
114 		}
115 
116 		this.map.events.on({
117 			buttonclick : this.onButtonClick,
118 			moveend : this.update,
119 			scope : this
120 		});
121 
122 		if (this.maximized) {
123 			this.maximizeControl();
124 		}
125 		return this.div;
126 	},
127 
128 	CLASS_NAME : 'OverviewMap'
129 });
