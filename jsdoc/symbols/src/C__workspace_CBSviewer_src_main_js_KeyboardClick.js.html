  7 /**
  8  * @class OpenLayers.Control.KeyboardClick
  9  * @requires OpenLayers/Control.js
 10  * @requires KeyboardPoint.js
 11  * 
 12  * A custom control that (a) adds a vector point that can be moved using the
 13  * arrow keys of the keyboard, and (b) displays a browser alert window when the
 14  * RETURN key is pressed. The control can be activated/deactivated using the "i"
 15  * key. When activated the control deactivates any KeyboardDefaults control in
 16  * the map so that the map is not moved when the arrow keys are pressed.
 17  * 
 18  * This control relies on the OpenLayers.Handler.KeyboardPoint custom handler.
 19  */
 20 OpenLayers.Control.KeyboardClick = OpenLayers.Class(OpenLayers.Control, {
 21 	initialize : function(options) {
 22 		OpenLayers.Control.prototype.initialize.apply(this, [ options ]);
 23 		var observeElement = this.observeElement || document;
 24 		this.handler = new OpenLayers.Handler.KeyboardPoint(this, {
 25 			done : this.onClick,
 26 			cancel : this.deactivate
 27 		}, {
 28 			observeElement : observeElement
 29 		});
 30 		OpenLayers.Event.observe(observeElement, "keydown", OpenLayers.Function.bindAsEventListener(function(evt) {
 31 			if (evt.keyCode == 73) { // "i"
 32 				if (this.active) {
 33 					this.deactivate();
 34 				} else {
 35 					this.activate();
 36 				}
 37 			}
 38 		}, this));
 39 	},
 40 
 41 	/**
 42 	 * Event handler voor click event.
 43 	 * 
 44 	 * @param geometry
 45 	 *            {Openlayers.Geometry}
 46 	 */
 47 	onClick : function(geometry) {
 48 		// TODO
 49 		alert("You clicked near " + geometry.x + " N, " + geometry.y + " E");
 50 	},
 51 
 52 	/**
 53 	 * Activeert deze control en zet de default keybord handler uit (mits
 54 	 * aanwezig).
 55 	 * 
 56 	 * @returns {Boolean}
 57 	 */
 58 	activate : function() {
 59 		if (!OpenLayers.Control.prototype.activate.apply(this, arguments)) {
 60 			return false;
 61 		}
 62 		// deactivate any KeyboardDefaults control
 63 		var keyboardDefaults = this.map.getControlsByClass('OpenLayers.Control.KeyboardDefaults')[0];
 64 		if (keyboardDefaults) {
 65 			keyboardDefaults.deactivate();
 66 		}
 67 		return true;
 68 	},
 69 
 70 	/**
 71 	 * Deactiveert deze control en zet de default keybord handler aan (mits
 72 	 * aanwezig).
 73 	 * 
 74 	 * @returns {Boolean}
 75 	 */
 76 	deactivate : function() {
 77 		if (!OpenLayers.Control.prototype.deactivate.apply(this, arguments)) {
 78 			return false;
 79 		}
 80 		// reactivate any KeyboardDefaults control
 81 		var keyboardDefaults = this.map.getControlsByClass('OpenLayers.Control.KeyboardDefaults')[0];
 82 		if (keyboardDefaults) {
 83 			keyboardDefaults.activate();
 84 		}
 85 		return true;
 86 	}
 87 });
