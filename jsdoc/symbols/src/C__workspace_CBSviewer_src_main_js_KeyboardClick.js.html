  7 /**
  8  * A custom control that (a) adds a vector point that can be moved using the
  9  * arrow keys of the keyboard, and (b) displays a browser alert window when the
 10  * RETURN key is pressed. The control can be activated/deactivated using the "i"
 11  * key. When activated the control deactivates any KeyboardDefaults control in
 12  * the map so that the map is not moved when the arrow keys are pressed.
 13  * 
 14  * This control relies on the OpenLayers.Handler.KeyboardPoint custom handler.
 15  * 
 16  * @class OpenLayers.Control.KeyboardClick
 17  * @requires OpenLayers/Control.js
 18  * @requires KeyboardPoint.js
 19  * @extends OpenLayers.Control
 20  */
 21 OpenLayers.Control.KeyboardClick = OpenLayers.Class(OpenLayers.Control, {
 22 	/** @constructor */
 23 	initialize : function(options) {
 24 		OpenLayers.Control.prototype.initialize.apply(this, [ options ]);
 25 		var observeElement = this.observeElement || document;
 26 		this.handler = new OpenLayers.Handler.KeyboardPoint(this, {
 27 			done : this.onClick,
 28 			cancel : this.deactivate
 29 		}, {
 30 			observeElement : observeElement
 31 		});
 32 		OpenLayers.Event.observe(observeElement, "keydown", OpenLayers.Function.bindAsEventListener(function(evt) {
 33 			if (evt.keyCode == 73) { // "i"
 34 				if (this.active) {
 35 					this.deactivate();
 36 				} else {
 37 					this.activate();
 38 				}
 39 			}
 40 		}, this));
 41 	},
 42 
 43 	/**
 44 	 * Event handler voor click event.
 45 	 * 
 46 	 * @param {Openlayers.Geometry}
 47 	 *            geometry met real life coordinaten
 48 	 * 
 49 	 */
 50 	onClick : function(geometry) {
 51 		var featureInfoCtl = this.map.getControlsByClass('WMSGetFeatureInfo')[0];
 52 		if (featureInfoCtl) {
 53 			var lonlat = new OpenLayers.LonLat([ geometry.x, geometry.y ]);
 54 			var pixel = this.map.getPixelFromLonLat(lonlat);
 55 			featureInfoCtl.getInfoForClick({
 56 				'xy' : pixel,
 57 				'lonlat' : lonlat
 58 			});
 59 		}
 60 	},
 61 
 62 	/**
 63 	 * Activeert deze control en zet de default keybord handler uit (mits
 64 	 * aanwezig).
 65 	 * 
 66 	 * @returns {Boolean}
 67 	 */
 68 	activate : function() {
 69 		if (!OpenLayers.Control.prototype.activate.apply(this, arguments)) {
 70 			return false;
 71 		}
 72 		// deactivate any KeyboardDefaults control
 73 		var keyboardDefaults = this.map.getControlsByClass('OpenLayers.Control.KeyboardDefaults')[0];
 74 		if (keyboardDefaults) {
 75 			keyboardDefaults.deactivate();
 76 		}
 77 		return true;
 78 	},
 79 
 80 	/**
 81 	 * Deactiveert deze control en zet de default keybord handler aan (mits
 82 	 * aanwezig).
 83 	 * 
 84 	 * @returns {Boolean}
 85 	 */
 86 	deactivate : function() {
 87 		if (!OpenLayers.Control.prototype.deactivate.apply(this, arguments)) {
 88 			return false;
 89 		}
 90 		// reactivate any KeyboardDefaults control
 91 		var keyboardDefaults = this.map.getControlsByClass('OpenLayers.Control.KeyboardDefaults')[0];
 92 		if (keyboardDefaults) {
 93 			keyboardDefaults.activate();
 94 		}
 95 		return true;
 96 	}
 97 });
