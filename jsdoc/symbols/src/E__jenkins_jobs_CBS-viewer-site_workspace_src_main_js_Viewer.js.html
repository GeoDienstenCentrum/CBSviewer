 60 	/**
 61 	 * zorgt voor correct afhandelen van viewport resize.
 62 	 * 
 63 	 * @private
 64 	 */
 65 	function _resize() {
 66 		if (_fullSize) {
 67 			var borderW = parseInt(jQuery('#' + this.config.mapDiv).css('borderLeftWidth'), 10)
 68 					+ parseInt(jQuery('#' + this.config.mapDiv).css('borderRightWidth'), 10);
 69 			var borderH = parseInt(jQuery('#' + this.config.mapDiv).css('borderTopWidth'), 10)
 70 					+ parseInt(jQuery('#' + this.config.mapDiv).css('borderBottomWidth'), 10);
 71 			// inhoud padding set in css
 72 			var headerH = parseInt(jQuery('#' + this.config.mapDiv).parent().parent().css('padding-top'), 10);
 73 			var footerH = parseInt(jQuery('#' + this.config.mapDiv).parent().parent().css('padding-bottom'), 10);
 74 
 75 			var w = jQuery('#' + this.config.mapDiv).parent().width() - borderW;
 76 			var h = jQuery(window).height() - headerH - footerH - borderH;
 77 
 78 			jQuery('#' + this.config.mapDiv).width(w).height(h);
 79 			_map.updateSize();
 80 			var vectors = _map.getLayersByClass("OpenLayers.Layer.Vector");
 81 			if (vectors.length > 0) {
 82 				// in dit geval is er een kaartlaag met een featureinfo lokatie
 83 				// verschuif naar die lokatie
 84 				var bounds = vectors[0].getDataExtent();
 85 				_map.panTo(bounds.getCenterLonLat());
 86 			}
 87 		}
 88 	}

 90 	/**
 91 	 * Stelt de doorzichtigheid van de voorgrond kaart in.
 92 	 * 
 93 	 * @param alpha
 94 	 *            float waarde tussen 0 1n 1
 95 	 * @private
 96 	 */
 97 	function _setOpacity(alpha) {
 98 		alpha = parseFloat(alpha);
 99 		if (0.09 < alpha && alpha < 0.91) {
100 			_opacity = alpha;
101 			var lyrs = _map.getLayersByClass('OpenLayers.Layer.WMS');
102 			for ( var lyr = 0; lyr < lyrs.length; lyr++) {
103 				if (!lyrs[lyr].isBaseLayer) {
104 					lyrs[lyr].setOpacity(_opacity);
105 				}
106 			}
107 		}
108 	}

341 		/**
342 		 * Voeg WMS toe aan de kaart. Uitgangspunt is dat de WMS transparante
343 		 * PNG ondersteund. eerder geladen WMS lagen worden verwijderd
344 		 * 
345 		 * @param {object}
346 		 *            wmsConfig Een object met WMS parameters. <code>
347 		 * {
348 		 * 'name' : 'cbs_inwoners_2010_per_hectare',
349 		 * 'url' : 'http://geodata.nationaalgeoregister.nl/cbsvierkanten100m2010/ows',
350 		 * 'layers' : 'cbsvierkanten100m2010:cbs_inwoners_2000_per_hectare',
351 		 * 'styles' : 'cbsvierkant100m_inwoners_2000'
352 		 * }
353 		 * </code>
354 		 */
355 		loadWMS : function(wmsConfig) {
356 			this.removeOverlays();
357 			var layer = new OpenLayers.Layer.WMS(wmsConfig.name, wmsConfig.url, {
358 				layers : wmsConfig.layers,
359 				styles : wmsConfig.styles,
360 				version : '1.3.0',
361 				format : 'image/png',
362 				transparent : true
363 			}, {
364 				isBaseLayer : false,
365 				visibility : true,
366 				singleTile : false,
367 				opacity : _opacity
368 			});
369 			_map.addLayer(layer);
370 			var fInfoControl = _map.getControlsByClass('WMSGetFeatureInfo');
371 			fInfoControl[0].url = wmsConfig.url;
372 		},

374 		/**
375 		 * verwijder de WMS uit de kaart (behalve als het een base layer is).
376 		 * 
377 		 * @param {string}
378 		 *            wmsLyrName naam van de WMS service
379 		 */
380 		removeWMS : function(wmsLyrName) {
381 			var lyrs = _map.getLayersByName(wmsLyrName);
382 			for ( var lyr = 0; lyr < lyrs.length; lyr++) {
383 				if (!lyrs[lyr].isBaseLayer) {
384 					_map.removeLayer(lyrs[lyr]);
385 					lyrs[lyr].destroy();
386 				}
387 			}
388 		},

390 		/**
391 		 * verwijder alle overlays. Voorlopig alleen type {OpenLayers.Layer.WMS}
392 		 */
393 		removeOverlays : function() {
394 			// reset featureinfo text
395 			jQuery('#' + config.featureInfoDiv).html(OpenLayers.i18n('KEY_INFO_GEEN_FEATURES'));
396 			// verwijder ikoontjes die de controls tekenen
397 			var lyrs = _map.getLayersByName('ClickDrawControl'), lyr;
398 			for (lyr = 0; lyr < lyrs.length; lyr++) {
399 				lyrs[lyr].removeAllFeatures();
400 			}
401 			lyrs = _map.getLayersByName('OpenLayers.Handler.KeyboardPoint');
402 			for (lyr = 0; lyr < lyrs.length; lyr++) {
403 				lyrs[lyr].removeAllFeatures();
404 			}
405 
406 			// verwijder WMS lagen
407 			lyrs = _map.getLayersByClass('OpenLayers.Layer.WMS');
408 			for (lyr = 0; lyr < lyrs.length; lyr++) {
409 				if (!lyrs[lyr].isBaseLayer) {
410 					_map.removeLayer(lyrs[lyr]);
411 					lyrs[lyr].destroy();
412 				}
413 			}
414 		},

441 				var borderH = parseInt(jQuery('#' + this.config.mapDiv).css('borderTopWidth'), 10)
442 						+ parseInt(jQuery('#' + this.config.mapDiv).css('borderBottomWidth'), 10);
443 				var headerH = parseInt(jQuery('#' + this.config.mapDiv).parent().parent().css('padding-top'), 10); // inhoud
444 				// padding
445 				// set
446 				// in
447 				// css
448 				var footerH = parseInt(jQuery('#' + this.config.mapDiv).parent().parent().css('padding-bottom'), 10);
449

471 		/**
472 		 * Toggle basemap.
473 		 */
474 		toggleBaseMap : function() {
475 			var topo = _map.getLayersByName('topo')[0];
476 			var lufo = _map.getLayersByName('lufo')[0];
477 			if (topo.getVisibility()) {
478 				_map.setBaseLayer(lufo);
479 				jQuery('#toggleBaseMap').text(OpenLayers.i18n('KEY_TOGGLE_BASEMAP_TOPO'));
480 			} else {
481 				_map.setBaseLayer(topo);
482 				jQuery('#toggleBaseMap').text(OpenLayers.i18n('KEY_TOGGLE_BASEMAP_LUFO'));
483 			}
484 			jQuery('#toggleBaseMap').toggleClass('lufo topo');
485 		},

488 		 * set up basemap.
489 		 * 
490 		 * @private

492 		addBaseMap : function() {

501 				layer : "brtachtergrondkaart",
502 				matrixSet : 'EPSG:28992',
503 				matrixIds : matrixIds,
504 				format : 'image/png8',
505 				style : '_null'
506 			}));
507 			// luchtfoto
508 			_map.addLayer(new OpenLayers.Layer.WMS(
509 					'lufo', 'http://gisdemo2.agro.nl/arcgis/services/Luchtfoto2010/MapServer/WMSServer?service=WMS', {
510 						layers : 0,
511 						version : '1.3.0',
512 						format : 'image/jpeg',
513 						transparent : false
514 					}, {
515 						isBaseLayer : true,
516 						visibility : true,
517 						singleTile : false
518 					}));
519 
520 		}
