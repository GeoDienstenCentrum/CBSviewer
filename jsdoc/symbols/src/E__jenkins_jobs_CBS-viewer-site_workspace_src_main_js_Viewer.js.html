  7 /**
  8  * Viewer.
  9  * 
 10  * @author mprins
 11  * @returns {Viewer} Viewer object
 12  * @class
 13  */
 14 Viewer = function() {
 15 	/**
 16 	 * Map object, initially null.
 17 	 * 
 18 	 * @private
 19 	 * @type {OpenLayers.Map}
 20 	 */
 21 	var _map = null;
 22 
 23 	/**
 24 	 * Toggle vlag voor fullsize functie.
 25 	 * 
 26 	 * @type {Boolean}
 27 	 * @private
 28 	 */
 29 	var _fullSize = false;
 30 	/**
 31 	 * Window resize timout flag.
 32 	 * 
 33 	 * @type {Boolean}
 34 	 * @private
 35 	 */
 36 	var _resizeTimeOut = false;
 37 
 38 	/**
 39 	 * update het informatie element met feature info.
 40 	 * 
 41 	 * @param {OpenLayers.Event}
 42 	 *            evt Het featureinfo event
 43 	 * @private
 44 	 */
 45 	function _showInfo(evt) {
 46 		if (evt.text !== undefined) {
 47 			jQuery('#' + config.featureInfoDiv).html(evt.text);
 48 			jQuery('#' + config.featureInfoDiv).change();
 49 		}
 50 	}
 51 
 52 	/**
 53 	 * zorgt voor correct afhandelen van viewport resize.
 54 	 * 
 55 	 * @private
 56 	 */
 57 	function _resize() {
 58 		if (_fullSize) {
 59 			var borderW = parseInt(jQuery('#' + this.config.mapDiv).css('borderLeftWidth'), 10)
 60 					+ parseInt(jQuery('#' + this.config.mapDiv).css('borderRightWidth'), 10);
 61 			var borderH = parseInt(jQuery('#' + this.config.mapDiv).css('borderTopWidth'), 10)
 62 					+ parseInt(jQuery('#' + this.config.mapDiv).css('borderBottomWidth'), 10);
 63 			var w = jQuery('#' + this.config.mapDiv).parent().width() - borderW;
 64 			// var h = jQuery('#' + this.config.mapDiv).parent().height() -
 65 			// borderH;
 66 			var h = jQuery('#' + this.config.mapDiv).parent().parent().height() - borderH;
 67 
 68 			jQuery('#' + this.config.mapDiv).width(w).height(h);
 69 			_map.updateSize();
 70 			var vectors = _map.getLayersByClass("OpenLayers.Layer.Vector");
 71 			if (vectors.length > 0) {
 72 				// in dit geval is er een kaartlaag met een featureinfo lokatie
 73 				// verschuif naar die lokatie
 74 				var bounds = vectors[0].getDataExtent();
 75 				_map.panTo(bounds.getCenterLonLat());
 76 			}
 77 		}
 78 	}
 79 	/**
 80 	 * Publieke interface van deze klasse.
 81 	 * 
 82 	 * @returns {Viewer} publieke methodes
 83 	 */
 84 	return {
 85 
 86 		/**
 87 		 * Constructor, attach to the DOM.
 88 		 * 
 89 		 * @param {object}
 90 		 *            config Configratie object
 91 		 */
 92 		init : function(config) {
 93 			this.config = config;
 94 			OpenLayers.ImgPath = config.imgPath;
 95 			OpenLayers.IMAGE_RELOAD_ATTEMPTS = 2;
 96 			OpenLayers.Number.decimalSeparator = ",";
 97 
 98 			// merge any controls met default
 99 			jQuery.extend(true, this.config, {
100 				map : {
101 					controls : []
102 				}
103 			});
104 			jQuery(window).unload(function() {
105 				Viewer.destroy();
106 			});
107 
108 			jQuery('#' + this.config.mapDiv).width(this.config.map.width).height(this.config.map.height);
109 
110 			_map = new OpenLayers.Map(this.config.mapDiv, this.config.map);
111 			this.addBaseMap();
112 			this.addControls();
113 			_map.zoomTo(this.config.map.initialZoom);
114 
115 			// toggle knop voor omschakelen basemap
116 			var aToggle = '<a class="lufo" href="#" id="toggleBaseMap" title="'
117 					+ OpenLayers.i18n('KEY_TOGGLE_BASEMAP_TITLE') + '" onclick="Viewer.toggleBaseMap();">'
118 					+ OpenLayers.i18n('KEY_TOGGLE_BASEMAP_LUFO') + '</a>';
119 			jQuery('#' + config.mapDiv).prepend(aToggle);
120 
121 			if (this.config.toggleSize) {
122 				// toggle knop voor vergroten/verkleinen van de kaart
123 				var aToggleMapSize = '<a class="max" href="#" id="toggleSize" title="'
124 						+ OpenLayers.i18n('KEY_TOGGLE_SIZE') + '" onclick="Viewer.toggleFullSize();"></a>';
125 				jQuery('#' + config.mapDiv).prepend(aToggleMapSize);
126 			}
127 			if (this.config.fullSize) {
128 				this.toggleFullSize();
129 			}
130 			jQuery(window).resize(function() {
131 				// aanhaken bij window resize
132 				if (_resizeTimeOut) {
133 					clearTimeout(_resizeTimeOut);
134 				}
135 				_resizeTimeOut = setTimeout(_resize, 200); // 200 is time in
136 				// miliseconds
137 			});
138 		},
139 
140 		/**
141 		 * Accessor voor de kaart.
142 		 * 
143 		 * @return {OpenLayers.Map} object van deze Viewer of null als het
144 		 *         object niet is geinitialiseerd
145 		 * @deprecated probeer deze niet te gebruiken
146 		 */
147 		getMap : function() {
148 			(window.console && console.warn('Deprecated method called: Viewer::getMap().'));
149 			return _map;
150 		},
151 
152 		/**
153 		 * verplaats en zoom de kaart naar de gevraagde locatie.
154 		 * 
155 		 * @param {number}
156 		 *            x de x coordinaat
157 		 * @param {number}
158 		 *            y de y coordinaat
159 		 * @param {number}
160 		 *            radius straal van het gebied
161 		 */
162 		zoomTo : function(x, y, radius) {
163 			_map.panTo(new OpenLayers.LonLat(x, y));
164 			var zm = _map.getZoomForExtent(new OpenLayers.Bounds(x - radius, y - radius, x + radius, y + radius));
165 			_map.zoomTo(zm);
166 		},
167 
168 		/**
169 		 * Controls aan de kaart hangen.
170 		 * 
171 		 * @private
172 		 */
173 		addControls : function() {
174 			_map.addControl(new UpdateLegendControl({
175 				div : jQuery('#' + this.config.legendDiv)[0]
176 			}));
177 			_map.addControl(new OpenLayers.Control.KeyboardDefaults({
178 				/* alleen actief als de kaart focus heeft */
179 				observeElement : this.config.mapDiv
180 			}));
181 			_map.addControl(new OpenLayers.Control.Zoom());
182 			_map.addControl(new OpenLayers.Control.Navigation({
183 				zoomWheelEnabled : false
184 			}));
185 			_map.addControl(new OpenLayers.Control.KeyboardClick({
186 				/* alleen actief als de kaart focus heeft */
187 				observeElement : this.config.mapDiv
188 			}));
189 			_map.addControl(new WMSGetFeatureInfo({
190 				eventListeners : {
191 					getfeatureinfo : _showInfo
192 				}
193 			}));
194 			_map.addControl(new ClickDrawControl());
195 		},
196 
197 		/**
198 		 * cleanup. Moet aangeroepen voor dat een eventueel DOM element van de
199 		 * pagina wordt verwijderd. Wordt automatische aangeroepen bij verlaten
200 		 * van de pagina.
201 		 */
202 		destroy : function() {
203 			if (_map !== null) {
204 				_map.destroy();
205 				_map = null;
206 			}
207 		},
208 
209 		/**
210 		 * Voeg WMS toe aan de kaart. Uitgangspunt is dat de WMS transparante
211 		 * PNG ondersteund. eerder geladen WMS lagen worden verwijderd
212 		 * 
213 		 * @param {object}
214 		 *            wmsConfig Een object met WMS parameters. <code>
215 		 * {
216 		 * 'name' : 'cbs_inwoners_2010_per_hectare',
217 		 * 'url' : 'http://geodata.nationaalgeoregister.nl/cbsvierkanten100m2010/ows',
218 		 * 'layers' : 'cbsvierkanten100m2010:cbs_inwoners_2000_per_hectare',
219 		 * 'styles' : 'cbsvierkant100m_inwoners_2000'
220 		 * }
221 		 * </code>
222 		 */
223 		loadWMS : function(wmsConfig) {
224 			this.removeOverlays();
225 			var layer = new OpenLayers.Layer.WMS(wmsConfig.name, wmsConfig.url, {
226 				layers : wmsConfig.layers,
227 				styles : wmsConfig.styles,
228 				version : '1.3.0',
229 				format : 'image/png',
230 				transparent : true
231 			}, {
232 				isBaseLayer : false,
233 				visibility : true,
234 				singleTile : false,
235 				opacity : 0.8
236 			});
237 			_map.addLayer(layer);
238 		},
239 
240 		/**
241 		 * verwijder de WMS uit de kaart (behalve als het een base layer is).
242 		 * 
243 		 * @param {string}
244 		 *            wmsLyrName naam van de WMS service
245 		 */
246 		removeWMS : function(wmsLyrName) {
247 			var lyrs = _map.getLayersByName(wmsLyrName);
248 			for ( var lyr = 0; lyr < lyrs.length; lyr++) {
249 				if (!lyrs[lyr].isBaseLayer) {
250 					_map.removeLayer(lyrs[lyr]);
251 					lyrs[lyr].destroy();
252 				}
253 			}
254 		},
255 
256 		/**
257 		 * verwijder alle overlays. Voorlopig alleen type {OpenLayers.Layer.WMS}
258 		 */
259 		removeOverlays : function() {
260 			var lyrs = _map.getLayersByClass('OpenLayers.Layer.WMS');
261 			for ( var lyr = 0; lyr < lyrs.length; lyr++) {
262 				if (!lyrs[lyr].isBaseLayer) {
263 					_map.removeLayer(lyrs[lyr]);
264 					lyrs[lyr].destroy();
265 				}
266 			}
267 		},
268 
269 		/**
270 		 * Afmeting van de kaart aanpassen aan schermbreedte.
271 		 * 
272 		 * @todo hoogte instellen werkt niet omdat de parent hoogte de afmeting
273 		 *       van de child heeft.
274 		 */
275 		toggleFullSize : function() {
276 			if (_fullSize) {
277 				// terugzetten
278 				jQuery('#' + this.config.mapDiv).width(this.config.map.width).height(this.config.map.height);
279 				jQuery('#toggleSize').toggleClass('restore max');
280 				_fullSize = false;
281 				_map.updateSize();
282 
283 				var vectors = _map.getLayersByClass("OpenLayers.Layer.Vector");
284 				if (vectors.length > 0) {
285 					// in dit geval is er een kaartlaag met een featureinfo
286 					// lokatie verschuif naar die lokatie
287 					var bounds = vectors[0].getDataExtent();
288 					_map.panTo(bounds.getCenterLonLat());
289 				}
290 			} else {
291 				// vergroten
292 				var borderW = parseInt(jQuery('#' + this.config.mapDiv).css('borderLeftWidth'), 10)
293 						+ parseInt(jQuery('#' + this.config.mapDiv).css('borderRightWidth'), 10);
294 				var borderH = parseInt(jQuery('#' + this.config.mapDiv).css('borderTopWidth'), 10)
295 						+ parseInt(jQuery('#' + this.config.mapDiv).css('borderBottomWidth'), 10);
296 
297 				var w = jQuery('#' + this.config.mapDiv).parent().width() - borderW;
298 				// var hh = jQuery('#' + this.config.mapDiv).parent().height();
299 				var h = jQuery('#' + this.config.mapDiv).parent().parent().height() - borderH;
300 
301 				jQuery('#' + this.config.mapDiv).width(w).height(h);
302 				jQuery('#toggleSize').toggleClass('restore max');
303 				_fullSize = true;
304 				_map.updateSize();
305 			}
306 		},
307 
308 		/**
309 		 * Stel de kaart in voor afdrukken.
310 		 * 
311 		 * @returns
312 		 */
313 		printPrepare : function() {
314 			if (_fullSize) {
315 				this.toggleFullSize();
316 			}
317 		},
318 
319 		/**
320 		 * Toggle basemap.
321 		 */
322 		toggleBaseMap : function() {
323 			var topo = _map.getLayersByName('topo')[0];
324 			var lufo = _map.getLayersByName('lufo')[0];
325 			if (topo.getVisibility()) {
326 				_map.setBaseLayer(lufo);
327 				jQuery('#toggleBaseMap').text(OpenLayers.i18n('KEY_TOGGLE_BASEMAP_TOPO'));
328 			} else {
329 				_map.setBaseLayer(topo);
330 				jQuery('#toggleBaseMap').text(OpenLayers.i18n('KEY_TOGGLE_BASEMAP_LUFO'));
331 			}
332 			jQuery('#toggleBaseMap').toggleClass('lufo topo');
333 		},
334 
335 		/**
336 		 * set up basemap.
337 		 * 
338 		 * @private
339 		 */
340 		addBaseMap : function() {
341 			// brtachtergrond
342 			var matrixIds = [ 13 ];
343 			for ( var i = 0; i < 13; ++i) {
344 				matrixIds[i] = "EPSG:28992:" + i;
345 			}
346 			_map.addLayer(new OpenLayers.Layer.WMTS({
347 				name : "topo",
348 				url : "http://geodata.nationaalgeoregister.nl/tiles/service/wmts/brtachtergrondkaart/",
349 				layer : "brtachtergrondkaart",
350 				matrixSet : 'EPSG:28992',
351 				matrixIds : matrixIds,
352 				format : 'image/png8',
353 				style : '_null'
354 			}));
355 			// luchtfoto
356 			_map.addLayer(new OpenLayers.Layer.WMS(
357 					'lufo', 'http://gisdemo2.agro.nl/arcgis/services/Luchtfoto2010/MapServer/WMSServer?service=WMS', {
358 						layers : 0,
359 						version : '1.3.0',
360 						format : 'image/jpeg',
361 						transparent : false
362 					}, {
363 						isBaseLayer : true,
364 						visibility : true,
365 						singleTile : false
366 					}));
367 
368 		}
369 	};
370 }();
